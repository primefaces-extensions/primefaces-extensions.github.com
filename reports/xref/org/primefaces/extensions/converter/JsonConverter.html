18  import java.io.Serializable;
19  import java.lang.reflect.Array;
20  import java.util.ArrayList;
21  import java.util.HashMap;
22  import java.util.List;
23  import java.util.Map;
24  
25  import javax.el.ValueExpression;
26  import javax.faces.application.FacesMessage;
27  import javax.faces.component.UIComponent;
28  import javax.faces.context.FacesContext;
29  import javax.faces.convert.Converter;
30  import javax.faces.convert.ConverterException;
31  import javax.faces.convert.FacesConverter;
32  
33  import org.apache.commons.lang3.StringUtils;
34  import org.primefaces.extensions.util.json.GsonConverter;
35  import org.primefaces.extensions.util.json.ParameterizedTypeImpl;
36  import org.primefaces.util.Constants;

38  /**
39   * {@link Converter} which converts a JSON string to an object an vice-versa.
40   *
41   * @author Thomas Andraschko / last modified by $Author$
42   * @version $Revision$
43   * @since 0.2
44   */
45  @FacesConverter(value = "org.primefaces.extensions.converter.JsonConverter")
46  public class JsonConverter implements Converter, Serializable {
47  
48      private static final long serialVersionUID = 20121214L;
49  
50      private static final Map<String, Class<?>> PRIMITIVE_CLASSES = new HashMap<>();
51      private static final Map<String, Class<?>> PRIMITIVE_ARRAY_CLASSES = new HashMap<>();
52  
53      static {
54          PRIMITIVE_CLASSES.put("boolean", boolean.class);
55          PRIMITIVE_CLASSES.put("byte", byte.class);
56          PRIMITIVE_CLASSES.put("short", short.class);
57          PRIMITIVE_CLASSES.put("char", char.class);
58          PRIMITIVE_CLASSES.put("int", int.class);
59          PRIMITIVE_CLASSES.put("long", long.class);
60          PRIMITIVE_CLASSES.put("float", float.class);
61          PRIMITIVE_CLASSES.put("double", double.class);
62  
63          PRIMITIVE_ARRAY_CLASSES.put("boolean[]", boolean[].class);
64          PRIMITIVE_ARRAY_CLASSES.put("byte[]", byte[].class);
65          PRIMITIVE_ARRAY_CLASSES.put("short[]", short[].class);
66          PRIMITIVE_ARRAY_CLASSES.put("char[]", char[].class);
67          PRIMITIVE_ARRAY_CLASSES.put("int[]", int[].class);
68          PRIMITIVE_ARRAY_CLASSES.put("long[]", long[].class);
69          PRIMITIVE_ARRAY_CLASSES.put("float[]", float[].class);
70          PRIMITIVE_ARRAY_CLASSES.put("double[]", double[].class);
71      }
72  
73      private String type;
74  
75      @Override
76      public Object getAsObject(FacesContext context, UIComponent component, String value) {
77          java.lang.reflect.Type objType;
78  
79          if (getType() == null) {
80              final ValueExpression expression = component.getValueExpression("value");
81              objType = expression.getType(context.getELContext());
82          }
83          else {
84              objType = getObjectType(getType().trim(), false);
85          }
86  
87          return GsonConverter.getGson().fromJson(value, objType);
88      }
89  
90      @Override
91      public String getAsString(FacesContext context, UIComponent component, Object value) {
92          if (getType() == null) {
93              return GsonConverter.getGson().toJson(value);
94          }
95          else {
96              return GsonConverter.getGson().toJson(value, getObjectType(getType().trim(), false));
97          }
98      }
99  
100     protected java.lang.reflect.Type getObjectType(String type, boolean isTypeArg) {
101         Class clazz = PRIMITIVE_CLASSES.get(type);
102         if (clazz != null) {
103             if (!isTypeArg) {
104                 return clazz;
105             }
106             else {
107                 throw new ConverterException(new FacesMessage(FacesMessage.SEVERITY_ERROR,
108                             "Type argument can not be a primitive type, but it was " + type
109                                         + ".",
110                             Constants.EMPTY_STRING));
111             }
112         }
113 
114         clazz = PRIMITIVE_ARRAY_CLASSES.get(type);
115         if (clazz != null) {
116             return clazz;
117         }
118 
119         final int arrayBracketIdx = type.indexOf('[');
120         final int leftBracketIdx = type.indexOf('<');
121         if (arrayBracketIdx >= 0 && (leftBracketIdx < 0 || arrayBracketIdx < leftBracketIdx)) {
122             // array
123             try {
124                 clazz = Class.forName(type.substring(0, arrayBracketIdx));
125 
126                 return Array.newInstance(clazz, 0).getClass();
127             }
128             catch (final ClassNotFoundException e) {
129                 throw new ConverterException(new FacesMessage(FacesMessage.SEVERITY_ERROR,
130                             "Class " + type.substring(0, arrayBracketIdx) + " not found",
131                             Constants.EMPTY_STRING));
132             }
133         }
134 
135         if (leftBracketIdx < 0) {
136             try {
137                 return Class.forName(type);
138             }
139             catch (final ClassNotFoundException e) {
140                 throw new ConverterException(new FacesMessage(FacesMessage.SEVERITY_ERROR, "Class " + type + " not found",
141                             Constants.EMPTY_STRING));
142             }
143         }
144 
145         final int rightBracketIdx = type.lastIndexOf('>');
146         if (rightBracketIdx < 0) {
147             throw new ConverterException(new FacesMessage(FacesMessage.SEVERITY_ERROR, type + " is not a valid generic type.",
148                         Constants.EMPTY_STRING));
149         }
150 
151         Class rawType;
152         try {
153             rawType = Class.forName(type.substring(0, leftBracketIdx));
154         }
155         catch (final ClassNotFoundException e) {
156             throw new ConverterException(new FacesMessage(FacesMessage.SEVERITY_ERROR,
157                         "Class " + type.substring(0, leftBracketIdx) + " not found",
158                         Constants.EMPTY_STRING));
159         }
160 
161         final String strTypeArgs = type.substring(leftBracketIdx + 1, rightBracketIdx);
162         final List<String> listTypeArgs = new ArrayList<>();
163         int startPos = 0;
164         int seekPos = 0;
165 
166         while (true) {
167             final int commaPos = strTypeArgs.indexOf(',', seekPos);
168             if (commaPos >= 0) {
169                 final String term = strTypeArgs.substring(startPos, commaPos);
170                 final int countLeftBrackets = StringUtils.countMatches(term, "<");
171                 final int countRightBrackets = StringUtils.countMatches(term, ">");
172                 if (countLeftBrackets == countRightBrackets) {
173                     listTypeArgs.add(term.trim());
174                     startPos = commaPos + 1;
175                 }
176 
177                 seekPos = commaPos + 1;
178             }
179             else {
180                 listTypeArgs.add(strTypeArgs.substring(startPos).trim());
181 
182                 break;
183             }
184         }
185 
186         if (listTypeArgs.isEmpty()) {
187             throw new ConverterException(new FacesMessage(FacesMessage.SEVERITY_ERROR, type + " is not a valid generic type.",
188                         Constants.EMPTY_STRING));
189         }
190 
191         final int size = listTypeArgs.size();
192         final java.lang.reflect.Type[] objectTypes = new java.lang.reflect.Type[size];
193         for (int i = 0; i < size; i++) {
194             // recursive call for each type argument
195             objectTypes[i] = getObjectType(listTypeArgs.get(i), true);
196         }
197 
198         return new ParameterizedTypeImpl(rawType, objectTypes, null);
199     }
200 
201     public String getType() {
202         return type;
203     }
204 
205     public void setType(String type) {
206         this.type = type;
207     }
208 }
