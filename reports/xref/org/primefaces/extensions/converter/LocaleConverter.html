16  package org.primefaces.extensions.converter;
17  
18  import java.io.Serializable;
19  import java.util.Locale;
20  
21  import javax.faces.application.FacesMessage;
22  import javax.faces.component.UIComponent;
23  import javax.faces.context.FacesContext;
24  import javax.faces.convert.Converter;
25  import javax.faces.convert.ConverterException;
26  import javax.faces.convert.FacesConverter;
27  
28  import org.apache.commons.lang3.ArrayUtils;
29  import org.apache.commons.lang3.StringUtils;
30  import org.primefaces.util.Constants;
31  import org.primefaces.util.LangUtils;
32  
33  /**
34   * {@link Converter} which converts a string to a {@link java.util.Locale} an vice-versa.
35   *
36   * @author Thomas Andraschko / last modified by $Author$
37   * @version $Revision$
38   * @since 0.2
39   */
40  @FacesConverter(value = "org.primefaces.extensions.converter.LocaleConverter")
41  public class LocaleConverter implements Converter, Serializable {
42  
43      private static final long serialVersionUID = 20121214L;
44  
45      private char separator = '_';
46  
47      @Override
48      public Object getAsObject(final FacesContext fc, final UIComponent component, final String value) {
49          if (LangUtils.isValueBlank(value)) {
50              return fc.getApplication().getDefaultLocale();
51          }
52  
53          return getLocaleObject(value, separator);
54      }
55  
56      @Override
57      public String getAsString(final FacesContext fc, final UIComponent component, final Object value) {
58          if (value == null) {
59              final Locale defaultLocale = fc.getApplication().getDefaultLocale();
60              if (defaultLocale == null) {
61                  return null;
62              }
63  
64              return getLocaleString(defaultLocale, separator);
65          }
66  
67          if (value instanceof String) {
68              return (String) value;
69          }
70          else if (value instanceof Locale) {
71              return getLocaleString((Locale) value, separator);
72          }
73          else {
74              throw new ConverterException(new FacesMessage(FacesMessage.SEVERITY_ERROR,
75                          "Wrong type: '" + value.getClass().getSimpleName()
76                                      + "' is not 'Locale'.",
77                          Constants.EMPTY_STRING));
78          }
79      }
80  
81      public static Locale getLocaleObject(final String locale, final char seperator) {
82          String replacedLocale = locale;
83          if (seperator != '-' && seperator != '_') {
84              replacedLocale = replacedLocale.replace(seperator, '_');
85          }
86  
87          replacedLocale = replacedLocale.replace('-', '_');
88  
89          final String[] parts = replacedLocale.split("_");
90          if (parts.length == 0
91                      || !parts[0].matches("[a-zA-Z]{2,2}")
92                      || parts.length > 1 && parts[1].length() != 0 && !parts[1].matches("[a-zA-Z]{2,2}")) {
93              throw new ConverterException(new FacesMessage(FacesMessage.SEVERITY_ERROR,
94                          "'" + locale + "' does not represent a valid locale",
95                          Constants.EMPTY_STRING));
96          }
97  
98          switch (parts.length) {
99              case 3:
100                 return new Locale(parts[0], parts[1], parts[2]);
101 
102             case 2:
103                 return new Locale(parts[0], parts[1]);
104 
105             case 1:
106                 return new Locale(parts[0]);
107 
108             default:
109                 return new Locale(parts[0], parts[1], StringUtils.join(ArrayUtils.subarray(parts, 2, parts.length), '_'));
110         }
111     }
112 
113     public static String getLocaleString(final Locale locale, final char seperator) {
114         if (LangUtils.isValueBlank(locale.getCountry())) {
115             return locale.getLanguage();
116         }
117 
118         return locale.getLanguage() + seperator + locale.getCountry();
119     }
120 
121     public char getSeparator() {
122         return separator;
123     }
124 
125     public void setSeparator(char separator) {
126         this.separator = separator;
127     }
128 }
