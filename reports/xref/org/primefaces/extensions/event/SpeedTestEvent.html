22  package org.primefaces.extensions.event;
23  
24  import javax.faces.component.UIComponent;
25  import javax.faces.component.behavior.Behavior;
26  
27  import org.primefaces.event.AbstractAjaxBehaviorEvent;
28  
29  /**
30   * Event which is triggered by the {@link org.primefaces.extensions.component.speedtest.Speedtest} component has the test's finished.
31   *
32   * @author ssibitz ssibitz@me.com
33   * @since 6.1
34   */
35  public class SpeedTestEvent extends AbstractAjaxBehaviorEvent {
36  
37      public static final String NAME = "speedtest";
38      private static final long serialVersionUID = 1L;
39  
40      private final Double pingTimeMS;
41  
42      private final Double jitterTimeMS;
43  
44      private final Double speedMbpsDownload;
45  
46      private final Double speedMbpsUpload;
47  
48      public SpeedTestEvent(final UIComponent component, final Behavior behavior,
49                  final Double _PingTimeMS, final Double _JitterTimeMS, final Double _SpeedMbpsDownload, final Double _SpeedMbpsUpload) {
50          super(component, behavior);
51          pingTimeMS = _PingTimeMS;
52          jitterTimeMS = _JitterTimeMS;
53          speedMbpsDownload = _SpeedMbpsDownload;
54          speedMbpsUpload = _SpeedMbpsUpload;
55      }
56  
57      public final Double getPingTimeMS() {
58          return pingTimeMS;
59      }
60  
61      public final Double getJitterTimeMS() {
62          return jitterTimeMS;
63      }
64  
65      public final Double getSpeedMbpsDownload() {
66          return speedMbpsDownload;
67      }
68  
69      public final Double getSpeedMbpsUpload() {
70          return speedMbpsUpload;
71      }
72  }
