16  package org.primefaces.extensions.application;
17  
18  import javax.faces.application.Resource;
19  import javax.faces.application.ResourceWrapper;
20  
21  /**
22   * {@link ResourceWrapper} which appends the version of PrimeFaces Extensions to the URL.
23   *
24   * @author Thomas Andraschko / last modified by $Author$
25   * @version $Revision$
26   * @since 0.1
27   */
28  public class PrimeFacesExtensionsResource extends ResourceWrapper {
29  
30      private final Resource wrapped;
31      private final String version;
32  
33      @SuppressWarnings("deprecation") // the default constructor is deprecated in JSF 2.3
34      public PrimeFacesExtensionsResource(final Resource resource) {
35          super();
36          wrapped = resource;
37          version = "&v=" + this.getClass().getPackage().getImplementationVersion();
38      }
39  
40      @Override
41      public Resource getWrapped() {
42          return wrapped;
43      }
44  
45      @Override
46      public String getRequestPath() {
47          return super.getRequestPath() + version;
48      }
49  
50      @Override
51      public String getContentType() {
52          return getWrapped().getContentType();
53      }
54  
55      @Override
56      public String getLibraryName() {
57          return getWrapped().getLibraryName();
58      }
59  
60      @Override
61      public String getResourceName() {
62          return getWrapped().getResourceName();
63      }
64  
65      @Override
66      public void setContentType(final String contentType) {
67          getWrapped().setContentType(contentType);
68      }
69  
70      @Override
71      public void setLibraryName(final String libraryName) {
72          getWrapped().setLibraryName(libraryName);
73      }
74  
75      @Override
76      public void setResourceName(final String resourceName) {
77          getWrapped().setResourceName(resourceName);
78      }
79  
80      @Override
81      public String toString() {
82          return getWrapped().toString();
83      }
84  }
