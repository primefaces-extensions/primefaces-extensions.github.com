22  /**

29  public class DynaFormControl extends AbstractDynaFormElement implements KeyData {
30  
31      public static final String DEFAULT_TYPE = "default";
32      private static final long serialVersionUID = 1L;
33      private static final String KEY_PREFIX_ROW = "r";
34      private static final String KEY_PREFIX_COLUMN = "c";
35      private static final String KEY_SUFFIX_REGULAR = "reg";
36      private static final String KEY_SUFFIX_EXTENDED = "ext";
37      private static final String KEY_SUFFIX_POSITION = "p";
38  
39      private String key;
40      private Object data;
41      private String type;
42      private int position;
43  
44      public DynaFormControl(Object data, String type, int colspan, int rowspan, int row, int column, int position,
45                  boolean extended) {
46          super(colspan, rowspan, row, column, extended);
47          this.position = position;
48  
49          this.data = data;
50          if (type != null) {
51              this.type = type;
52          }
53          else {
54              this.type = DEFAULT_TYPE;
55          }
56  
57          generateKey();
58      }
59

71      public Object getData() {

73      }
74  
75      @Override
76      public void setData(Object data) {
77          this.data = data;
78      }
79  
80      public String getType() {
81          return type;
82      }
83  
84      public int getPosition() {
85          return position;
86      }
87  
88      void setPosition(int position) {
89          this.position = position;
90      }
91  
92      @Override
93      public int hashCode() {
94          final int prime = 31;

98          return result;
99      }

122     void generateKey() {
123         final StringBuilder sb = new StringBuilder();
124         sb.append(KEY_PREFIX_ROW).append(getRow()).append(KEY_PREFIX_COLUMN).append(getColumn()).append(KEY_SUFFIX_POSITION).append(getPosition());
125         if (isExtended()) {
126             sb.append(KEY_SUFFIX_EXTENDED);
127         }
128         else {
129             sb.append(KEY_SUFFIX_REGULAR);
130         }
131         setKey(sb.toString());
132     }
133 
134     @Override
135     public String toString() {
136         return new ToStringBuilder(this, ToStringStyle.SHORT_PREFIX_STYLE).append("key", key).append("data", data)
137                     .append("type", type).append("colspan", getColspan())
138                     .append("rowspan", getRowspan()).append("row", getRow())
139                     .append("column", getColumn())
140                     .append("extended", isExtended())
141                     .append("position", getPosition()).toString();
142     }
