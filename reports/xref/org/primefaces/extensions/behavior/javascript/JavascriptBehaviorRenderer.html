16  package org.primefaces.extensions.behavior.javascript;
17  
18  import java.util.Collection;
19  
20  import javax.faces.component.UIComponent;
21  import javax.faces.component.UIParameter;
22  import javax.faces.component.behavior.ClientBehavior;
23  import javax.faces.component.behavior.ClientBehaviorContext;
24  import javax.faces.context.FacesContext;
25  import javax.faces.render.ClientBehaviorRenderer;
26  
27  import org.primefaces.component.api.ClientBehaviorRenderingMode;
28  
29  /**
30   * {@link ClientBehaviorRenderer} implementation for the {@link JavascriptBehavior}.
31   *
32   * @author Thomas Andraschko / last modified by $Author$
33   * @version $Revision$
34   * @since 0.2
35   */
36  public class JavascriptBehaviorRenderer extends ClientBehaviorRenderer {
37  
38      @Override
39      public String getScript(final ClientBehaviorContext behaviorContext, final ClientBehavior behavior) {
40          final JavascriptBehavior javascriptBehavior = (JavascriptBehavior) behavior;
41          if (javascriptBehavior.isDisabled()) {
42              return null;
43          }
44  
45          final FacesContext context = behaviorContext.getFacesContext();
46          final UIComponent component = behaviorContext.getComponent();
47          String source = behaviorContext.getSourceId();
48          if (source == null) {
49              source = component.getClientId(context);
50          }
51  
52          final StringBuilder script = new StringBuilder();
53          script.append("PrimeFacesExt.behavior.Javascript({");
54          script.append("source:'").append(source).append("'");
55          script.append(",event:'").append(behaviorContext.getEventName()).append("'");
56          script.append(",execute:function(source,event,params,ext){");
57          script.append(javascriptBehavior.getExecute()).append(";}");
58  
59          // params
60          boolean paramWritten = false;
61  
62          for (final UIComponent child : component.getChildren()) {
63              if (child instanceof UIParameter) {
64                  final UIParameter parameter = (UIParameter) child;
65  
66                  if (paramWritten) {
67                      script.append(",");
68                  }
69                  else {
70                      paramWritten = true;
71                      script.append(",params:{");
72                  }
73  
74                  script.append("'");
75                  script.append(parameter.getName()).append("':'").append(parameter.getValue());
76                  script.append("'");
77              }
78          }
79  
80          if (paramWritten) {
81              script.append("}");
82          }
83  
84          ClientBehaviorRenderingMode renderingMode = null;
85          Collection<ClientBehaviorContext.Parameter> behaviorParameters = behaviorContext.getParameters();
86  
87          if (behaviorParameters != null && !behaviorParameters.isEmpty()) {
88              for (ClientBehaviorContext.Parameter behaviorParameter : behaviorParameters) {
89                  if (behaviorParameter.getValue() != null && behaviorParameter.getValue() instanceof ClientBehaviorRenderingMode) {
90                      renderingMode = (ClientBehaviorRenderingMode) behaviorParameter.getValue();
91                      break;
92                  }
93              }
94          }
95  
96          if (ClientBehaviorRenderingMode.UNOBSTRUSIVE.equals(renderingMode)) {
97              script.append("},ext);");
98          }
99          else {
100             script.append("});");
101         }
102 
103         return script.toString();
104     }
105 }
