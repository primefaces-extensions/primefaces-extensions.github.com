16  package org.primefaces.extensions.util.json;
17  
18  import java.lang.reflect.Type;
19  import java.util.Date;
20  
21  import com.google.gson.JsonDeserializationContext;
22  import com.google.gson.JsonDeserializer;
23  import com.google.gson.JsonElement;
24  import com.google.gson.JsonParseException;
25  import com.google.gson.JsonPrimitive;
26  import com.google.gson.JsonSerializationContext;
27  import com.google.gson.JsonSerializer;
28  
29  /**
30   * Gson Serializer / Deserializer for {@link Date}.
31   *
32   * @author Oleg Varaksin / last modified by $Author$
33   * @version $Revision$
34   * @since 0.6.2
35   */
36  public class DateTypeAdapter implements JsonSerializer<Date>, JsonDeserializer<Date> {
37  
38      public JsonElement serialize(Date src, Type typeOfSrc, JsonSerializationContext context) {
39          return new JsonPrimitive(src.getTime());
40      }
41  
42      public Date deserialize(JsonElement json, Type typeOfT, JsonDeserializationContext context) throws JsonParseException {
43          return new Date(json.getAsJsonPrimitive().getAsLong());
44      }
45  }
