16  package org.primefaces.extensions.util;
17  
18  import java.io.UnsupportedEncodingException;
19  import java.net.URLEncoder;
20  import java.util.logging.Logger;
21  
22  import javax.faces.context.FacesContext;
23  import javax.servlet.http.HttpServletRequest;
24  
25  import org.primefaces.extensions.converter.JsonConverter;
26  import org.primefaces.extensions.converter.JsonExposeAwareConverter;
27  
28  /**
29   * Builder for request parameters.
30   *
31   * @author Oleg Varaksin / last modified by $Author$
32   * @version $Revision$
33   * @since 1.1.0
34   */
35  public class RequestParameterBuilder {
36  
37      private static final Logger LOG = Logger.getLogger(RequestParameterBuilder.class.getName());
38  
39      private StringBuilder buffer;
40      private String originalUrl;
41      private JsonConverter jsonConverter;
42      private String encoding;
43      private boolean added;
44  
45      /**
46       * Creates a builder instance. This constructor is useful when we only use encode() and encodeJson() methods.
47       */
48      public RequestParameterBuilder() {
49          this(null);
50      }
51  
52      /**
53       * Creates a builder instance wihout URL or with the current request URL.
54       *
55       * @param useCurrentRequest boolean flag if the current request URL should be used or not
56       */
57      public RequestParameterBuilder(boolean useCurrentRequest) {
58          this(useCurrentRequest
59                      ? ((HttpServletRequest) FacesContext.getCurrentInstance().getExternalContext().getRequest()).getRequestURL()
60                                  .toString()
61                      : null);
62      }
63  
64      /**
65       * Creates a builder instance by the given URL.
66       *
67       * @param url URL
68       */
69      public RequestParameterBuilder(String url) {
70          this(url, null);
71      }
72  
73      /**
74       * Creates a builder instance by the given URL.
75       *
76       * @param url URL
77       * @param jsonConverter specific JsonConverter. It should extends {@link JsonConverter} from PrimeFaces Extensions. If null, the default implementation
78       *            {@link JsonExposeAwareConverter} is used.
79       */
80      public RequestParameterBuilder(String url, JsonConverter jsonConverter) {
81          buffer = new StringBuilder((url != null) ? url : org.primefaces.util.Constants.EMPTY_STRING);
82          originalUrl = (url != null) ? url : org.primefaces.util.Constants.EMPTY_STRING;
83  
84          this.jsonConverter = jsonConverter == null ? new JsonExposeAwareConverter(false) : jsonConverter;
85  
86          encoding = FacesContext.getCurrentInstance().getExternalContext().getRequestCharacterEncoding();
87          if (encoding == null) {
88              encoding = "UTF-8";
89          }
90      }
91  
92      /**
93       * Adds a request parameter to the URL without specifying a data type of the given parameter value. Parameter's value is converted to JSON notation when
94       * adding. Furthermore, it will be encoded according to the acquired encoding.
95       *
96       * @param name name of the request parameter
97       * @param value value of the request parameter
98       * @return RequestParameterBuilder updated this instance which can be reused
99       * @throws UnsupportedEncodingException DOCUMENT_ME
100      */
101     public RequestParameterBuilder paramJson(String name, Object value) throws UnsupportedEncodingException {
102         return paramJson(name, value, null);
103     }
104 
105     /**
106      * Adds a request parameter to the URL with specifying a data type of the given parameter value. Data type is sometimes required, especially for Java
107      * generic types, because type information is erased at runtime and the conversion to JSON will not work properly. Parameter's value is converted to JSON
108      * notation when adding. Furthermore, it will be encoded according to the acquired encoding.
109      *
110      * @param name name of the request parameter
111      * @param value value of the request parameter
112      * @param type data type of the value object. Any primitive type, array, non generic or generic type is supported. Data type is sometimes required to
113      *            convert a value to a JSON representation. All data types should be fully qualified. Examples: "boolean" "int" "long[]" "java.lang.String"
114      *            "java.util.Date" "java.util.Collection<java.lang.Integer>" "java.util.Map<java.lang.String, com.durr.FooPair<java.lang.Integer,
115      *            java.util.Date>>" "com.durr.FooNonGenericClass" "com.durr.FooGenericClass<java.lang.String, java.lang.Integer>"
116      *            "com.durr.FooGenericClass<int[], com.durr.FooGenericClass<com.durr.FooNonGenericClass, java.lang.Boolean>>".
117      * @return RequestParameterBuilder updated this instance which can be reused
118      * @throws UnsupportedEncodingException DOCUMENT_ME
119      */
120     public RequestParameterBuilder paramJson(String name, Object value, String type) throws UnsupportedEncodingException {
121         String encodedJsonValue = encodeJson(value, type);
122 
123         if (added || originalUrl.contains("?")) {
124             buffer.append("&");
125         }
126         else {
127             buffer.append("?");
128         }
129 
130         buffer.append(name);
131         buffer.append("=");
132         buffer.append(encodedJsonValue);
133 
134         // set a flag that at least one request parameter was added
135         added = true;
136 
137         return this;
138     }
139 
140     /**
141      * Adds a request parameter to the URL. This is a convenient method for primitive, plain data types. Parameter's value will not be converted to JSON
142      * notation when adding. It will be only encoded according to the acquired encoding. Note: null values will not be added.
143      *
144      * @param name name of the request parameter
145      * @param value value of the request parameter
146      * @return RequestParameterBuilder updated this instance which can be reused
147      * @throws UnsupportedEncodingException DOCUMENT_ME
148      */
149     public RequestParameterBuilder param(String name, Object value) throws UnsupportedEncodingException {
150         String encodedValue = encode(value);
151 
152         if (encodedValue == null) {
153             return this;
154         }
155 
156         if (added || originalUrl.contains("?")) {
157             buffer.append("&");
158         }
159         else {
160             buffer.append("?");
161         }
162 
163         buffer.append(name);
164         buffer.append("=");
165         buffer.append(encodedValue);
166 
167         // set a flag that at least one request parameter was added
168         added = true;
169 
170         return this;
171     }
172 
173     /**
174      * Encodes given value with a proper encoding. This is a convenient method for primitive, plain data types. Value will not be converted to JSON. Note: Value
175      * can be null.
176      *
177      * @param value value to be encoded
178      * @return String encoded value
179      * @throws UnsupportedEncodingException DOCUMENT_ME
180      */
181     public String encode(Object value) throws UnsupportedEncodingException {
182         if (value == null) {
183             return null;
184         }
185 
186         return URLEncoder.encode(value.toString(), encoding);
187     }
188 
189     /**
190      * Convertes give value to JSON and encodes the converted value with a proper encoding. Data type is sometimes required, especially for Java generic types,
191      * because type information is erased at runtime and the conversion to JSON will not work properly.
192      *
193      * @param value value to be converted and encoded
194      * @param type data type of the value object. Any primitive type, array, non generic or generic type is supported. Data type is sometimes required to
195      *            convert a value to a JSON representation. All data types should be fully qualified. Examples: "boolean" "int" "long[]" "java.lang.String"
196      *            "java.util.Date" "java.util.Collection<java.lang.Integer>" "java.util.Map<java.lang.String, com.durr.FooPair<java.lang.Integer,
197      *            java.util.Date>>" "com.durr.FooNonGenericClass" "com.durr.FooGenericClass<java.lang.String, java.lang.Integer>"
198      *            "com.durr.FooGenericClass<int[], com.durr.FooGenericClass<com.durr.FooNonGenericClass, java.lang.Boolean>>".
199      * @return String converted and encoded value
200      * @throws UnsupportedEncodingException DOCUMENT_ME
201      */
202     public String encodeJson(Object value, String type) throws UnsupportedEncodingException {
203         jsonConverter.setType(type);
204 
205         String jsonValue;
206         if (value == null) {
207             jsonValue = "null";
208         }
209         else {
210             jsonValue = jsonConverter.getAsString(null, null, value);
211         }
212 
213         return URLEncoder.encode(jsonValue, encoding);
214     }
215 
216     /**
217      * Convertes give value to JSON without to know the data type and encodes the converted value with a proper encoding.
218      *
219      * @param value value to be converted and encoded
220      * @return String converted and encoded value
221      * @throws UnsupportedEncodingException DOCUMENT_ME
222      */
223     public String encodeJson(Object value) throws UnsupportedEncodingException {
224         return encodeJson(value, null);
225     }
226 
227     /**
228      * Builds the end result.
229      *
230      * @return String end result
231      */
232     public String build() {
233         String url = buffer.toString();
234 
235         if (url.length() > 2083) {
236             LOG.warning("URL " + url + " is longer than 2083 chars (" + buffer.length()
237                         + "). It may not work properly in old IE versions.");
238         }
239 
240         return url;
241     }
242 
243     /**
244      * Resets the internal state in order to be reused.
245      *
246      * @return RequestParameterBuilder reseted builder
247      */
248     public RequestParameterBuilder reset() {
249         buffer = new StringBuilder(originalUrl);
250         jsonConverter.setType(null);
251         added = false;
252 
253         return this;
254     }
255 }
