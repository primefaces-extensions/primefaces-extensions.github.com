16  package org.primefaces.extensions.component.letteravatar;
17  
18  import java.io.IOException;
19  
20  import javax.faces.component.UIComponent;
21  import javax.faces.context.FacesContext;
22  import javax.faces.context.ResponseWriter;
23  import javax.faces.render.FacesRenderer;
24  
25  import org.primefaces.renderkit.CoreRenderer;
26  import org.primefaces.util.WidgetBuilder;
27  
28  /**
29   * Renderer for the {@link LetterAvatar} component.
30   *
31   * @author https://github.com/aripddev
32   * @since 7.0
33   */
34  @FacesRenderer(componentFamily = LetterAvatar.COMPONENT_FAMILY, rendererType = LetterAvatar.DEFAULT_RENDERER)
35  public class LetterAvatarRenderer extends CoreRenderer {
36  
37      @Override
38      public void encodeEnd(final FacesContext context, final UIComponent component) throws IOException {
39          if (context == null) {
40              throw new NullPointerException("No context defined!");
41          }
42  
43          final LetterAvatar letterAvatar = (LetterAvatar) component;
44  
45          encodeMarkup(context, letterAvatar);
46          encodeScript(context, letterAvatar);
47      }
48  
49      public void encodeMarkup(FacesContext context, LetterAvatar letterAvatar) throws IOException {
50          final ResponseWriter writer = context.getResponseWriter();
51  
52          final Integer width = letterAvatar.getWidth();
53          final Integer height = letterAvatar.getHeight();
54          final String avatar = letterAvatar.getValue();
55          final Boolean rounded = letterAvatar.isRounded();
56  
57          final String clientId = letterAvatar.getClientId(context);
58          final String style = letterAvatar.getStyle();
59          String styleClass = letterAvatar.getStyleClass();
60          styleClass = styleClass == null ? LetterAvatar.COMPONENT_CLASS : LetterAvatar.COMPONENT_CLASS + " " + styleClass;
61  
62          writer.startElement("img", letterAvatar);
63          writer.writeAttribute("id", clientId, null);
64  
65          if (rounded) {
66              styleClass = styleClass + " " + LetterAvatar.COMPONENT_CLASS_ROUNDED;
67          }
68          writer.writeAttribute("class", styleClass, "styleClass");
69  
70          if (style != null) {
71              writer.writeAttribute("style", style, "style");
72          }
73  
74          writer.writeAttribute("width", width, null);
75          writer.writeAttribute("height", height, null);
76          writer.writeAttribute("avatar", avatar, null);
77  
78          writer.endElement("img");
79      }
80  
81      private void encodeScript(final FacesContext context, final LetterAvatar letterAvatar) throws IOException {
82          final WidgetBuilder wb = getWidgetBuilder(context);
83          wb.init("ExtLetterAvatar", letterAvatar.resolveWidgetVar(), letterAvatar.getClientId(context));
84          encodeClientBehaviors(context, letterAvatar);
85          wb.finish();
86      }
87  
88  }
