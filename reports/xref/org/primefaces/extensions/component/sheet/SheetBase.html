791     /**
792      * Gets the rowKey for the current row
793      *
794      * @param context the faces context
795      * @return a row key value or null if the expression is not set
796      */
797     protected Object getRowKeyValue(final FacesContext context) {
798         final ValueExpression veRowKey = getValueExpression(PropertyKeys.rowKey.name());
799         if (veRowKey == null) {
800             throw new RuntimeException("RowKey required on sheet!");
801         }
802         final Object value = veRowKey.getValue(context.getELContext());
803         if (value == null) {
804             throw new RuntimeException("RowKey must resolve to non-null value for updates to work properly");
805         }
806         return value;
807     }

847     /**
848      * Update the sort direction
849      *
850      * @param sortOrder
851      */
852     public void setSortOrder(final java.lang.String sortOrder) {
853         // when updating, make sure we store off the original so it may be
854         // restored
855         final String orig = (String) getStateHelper().get(PropertyKeys.origSortOrder);
856         if (orig == null) {
857             // do not call getSortOrder as it defaults to ascending, we want
858             // null
859             // if this is the first call and there is no previous value.
860             getStateHelper().put(PropertyKeys.origSortOrder, getStateHelper().eval(PropertyKeys.sortOrder));
861         }
862         getStateHelper().put(PropertyKeys.sortOrder, sortOrder);
863     }

865     /**
866      * The error message to display when the sheet is in error.
867      *
868      * @return
869      */
870     public String getErrorMessage() {
871         final Object result = getStateHelper().eval(PropertyKeys.errorMessage);
872         if (result == null) {
873             return null;
874         }
875         return result.toString();
876     }
