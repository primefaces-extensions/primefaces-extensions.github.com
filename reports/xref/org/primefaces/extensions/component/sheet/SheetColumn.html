614     /**
615      * Process all validators (skip normal UIInput behavior)
616      */
617     @Override
618     public void validate(final FacesContext context) {
619         if (context == null) {
620             throw new NullPointerException();
621         }
622 
623         final Validator[] validators = getValidators();
624         final Object value = getValue();
625 
626         if (!validateRequired(context, value)) {
627             return;
628         }
629 
630         if (validators == null) {
631             return;
632         }
633 
634         for (final Validator validator : validators) {
635             try {
636                 validator.validate(context, this, value);
637             }
638             catch (final ValidatorException ve) {
639                 // If the validator throws an exception, we're
640                 // invalid, and we need to add a message
641                 setValid(false);
642                 FacesMessage message;
643                 final String validatorMessageString = getValidatorMessage();
644 
645                 if (null != validatorMessageString) {
646                     message = new FacesMessage(FacesMessage.SEVERITY_ERROR,
647                                 validatorMessageString,
648                                 validatorMessageString);
649                     message.setSeverity(FacesMessage.SEVERITY_ERROR);
650                 }
651                 else {
652                     final Collection<FacesMessage> messages = ve.getFacesMessages();
653                     if (null != messages) {
654                         message = null;
655                         final String cid = getClientId(context);
656                         for (final FacesMessage m : messages) {
657                             context.addMessage(cid, m);
658                         }
659                     }
660                     else {
661                         message = ve.getFacesMessage();
662                     }
663                 }
664                 if (message != null) {
665                     final Sheet sheet = getSheet();
666                     if (sheet == null) {
667                         return;
668                     }
669                     context.addMessage(getClientId(context), message);
670                     sheet.getInvalidUpdates().add(
671                                 new SheetInvalidUpdate(sheet.getRowKeyValue(context), sheet.getColumns().indexOf(this), this, value,
672                                             message.getDetail()));
673 
674                 }
675             }
676         }
677 
678     }

680     /**
681      * Validates the value against the required flags on this column.
682      *
683      * @param context the FacesContext
684      * @param newValue the new value for this column
685      * @return true if passes validation, otherwise valse
686      */
687     protected boolean validateRequired(final FacesContext context, final Object newValue) {
688         // If our value is valid, enforce the required property if present
689         if (isValid() && isRequired() && isEmpty(newValue)) {
690             final String requiredMessageStr = getRequiredMessage();
691             FacesMessage message;
692             if (null != requiredMessageStr) {
693                 message = new FacesMessage(FacesMessage.SEVERITY_ERROR,
694                             requiredMessageStr,
695                             requiredMessageStr);
696             }
697             else {
698                 message = new FacesMessage(FacesMessage.SEVERITY_ERROR,
699                             MESSAGE_REQUIRED,
700                             MESSAGE_REQUIRED);
701             }
702             context.addMessage(getClientId(context), message);
703             final Sheet sheet = getSheet();
704             if (sheet != null) {
705                 sheet.getInvalidUpdates().add(
706                             new SheetInvalidUpdate(sheet.getRowKeyValue(context), sheet.getColumns().indexOf(this), this, newValue,
707                                         message.getDetail()));
708             }
709             setValid(false);
710             return false;
711         }
712         return true;
713     }
