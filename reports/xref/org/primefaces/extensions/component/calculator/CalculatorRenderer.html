16  package org.primefaces.extensions.component.calculator;
17  
18  import java.io.IOException;
19  
20  import javax.faces.FacesException;
21  import javax.faces.component.UIComponent;
22  import javax.faces.component.UIInput;
23  import javax.faces.context.FacesContext;
24  
25  import org.apache.commons.lang3.StringUtils;
26  import org.primefaces.expression.SearchExpressionFacade;
27  import org.primefaces.renderkit.CoreRenderer;
28  import org.primefaces.util.ComponentUtils;
29  import org.primefaces.util.WidgetBuilder;
30  
31  /**
32   * Renderer for the {@link Calculator} component.
33   *
34   * @author Melloware mellowaredev@gmail.com
35   * @since 6.1
36   */
37  public class CalculatorRenderer extends CoreRenderer {
38  
39      @Override
40      public void decode(final FacesContext context, final UIComponent component) {
41          decodeBehaviors(context, component);
42      }
43  
44      @Override
45      public void encodeEnd(final FacesContext context, final UIComponent component) throws IOException {
46          final Calculator calculator = (Calculator) component;
47          encodeScript(context, calculator);
48      }
49  
50      private void encodeScript(final FacesContext context, final Calculator calculator) throws IOException {
51          String target = SearchExpressionFacade.resolveClientIds(context, calculator, calculator.getFor());
52          if (isValueBlank(target)) {
53              target = calculator.getParent().getClientId(context);
54          }
55  
56          final UIComponent targetComponent = SearchExpressionFacade.resolveComponent(context, calculator, target);
57          if (!(targetComponent instanceof UIInput)) {
58              throw new FacesException("Calculator must use for=\"target\" or be nested inside an input!");
59          }
60  
61          final WidgetBuilder wb = getWidgetBuilder(context);
62          wb.init("ExtCalculator", calculator.resolveWidgetVar(), calculator.getClientId(context));
63          wb.attr("target", target);
64          wb.attr("showOn", StringUtils.lowerCase(calculator.getShowOn()));
65          wb.attr("layout", StringUtils.lowerCase(calculator.getLayout()));
66          wb.attr("precision", calculator.getPrecision());
67          wb.attr("locale", calculator.calculateLocale().toString());
68          wb.attr("isRTL", ComponentUtils.isRTL(context, calculator));
69          wb.attr("calculatorClass", calculator.getStyleClass());
70  
71          if (calculator.getOnopen() != null) {
72              // Define a callback function before the panel is opened
73              wb.callback("onOpen", "function(value, inst)", calculator.getOnopen());
74          }
75          if (calculator.getOnclose() != null) {
76              // Define a callback function when the panel is closed
77              wb.callback("onClose", "function(value, inst)", calculator.getOnclose());
78          }
79          if (calculator.getOnbutton() != null) {
80              // Define a callback function when a button is activated
81              wb.callback("onButton", "function(label, value, inst)", calculator.getOnbutton());
82          }
83  
84          encodeClientBehaviors(context, calculator);
85  
86          wb.finish();
87      }
88  
89  }
