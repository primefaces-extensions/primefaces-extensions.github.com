19  import java.io.UnsupportedEncodingException;

22  import java.util.List;
23  
24  import javax.faces.application.Resource;
25  import javax.faces.application.ResourceHandler;
26  import javax.faces.component.UIComponent;

37  /**
38   * Renderer for the {@link DocumentViewer} component.
39   *
40   * @author f.strazzullo
41   * @author Melloware mellowaredev@gmail.com
42   * @since 3.0.0
43   */
44  public class DocumentViewerRenderer extends CoreRenderer {
45  
46      @Override
47      public void encodeEnd(final FacesContext context, final UIComponent component) throws IOException {
48          final DocumentViewer documentViewer = (DocumentViewer) component;
49          encodeMarkup(context, documentViewer);
50      }
51  
52      private void encodeMarkup(final FacesContext context, final DocumentViewer documentViewer) throws IOException {
53          // Section 508 frame title for assisted technology
54          String title = documentViewer.getTitle() != null ? documentViewer.getTitle() : documentViewer.getName();
55          title = StringUtils.defaultString(title, "Document Viewer");
56  
57          final ResponseWriter writer = context.getResponseWriter();
58          writer.startElement("iframe", documentViewer);
59          writer.writeAttribute("id", documentViewer.getClientId(), null);
60          writer.writeAttribute("style", documentViewer.getStyle(), null);
61          writer.writeAttribute("title", title, null);
62          writer.writeAttribute("width", documentViewer.getWidth() != null ? documentViewer.getWidth() : "100%", null);
63          writer.writeAttribute("height", documentViewer.getHeight(), null);
64          writer.writeAttribute("allowfullscreen", Constants.EMPTY_STRING, null);
65          writer.writeAttribute("webkitallowfullscreen", Constants.EMPTY_STRING, null);
66          writer.writeAttribute("src", generateSrc(context, documentViewer), null);
67          writer.endElement("iframe");
68      }
69  
70      private String generateSrc(final FacesContext context, final DocumentViewer documentViewer) throws IOException {
71          String imageSrc;
72          try {
73              imageSrc = URLEncoder.encode(getDocumentSource(context, documentViewer), "UTF-8");
74          }
75          catch (final Exception ex) {
76              throw new IOException(ex);
77          }
78  
79          final StringBuilder srcBuilder = new StringBuilder();
80          srcBuilder.append(getResourceURL(context));
81          srcBuilder.append("&file=");
82          srcBuilder.append(imageSrc);
83          srcBuilder.append(generateHashString(documentViewer));
84  
85          return srcBuilder.toString();
86      }
87  
88      private String generateHashString(final DocumentViewer documentViewer) {
89          final List<String> params = new ArrayList<>(4);
90          params.add("locale=" + documentViewer.calculateLocale().toString().replaceAll("_", "-"));
91  
92          // page: page number. Example: page=2
93          if (documentViewer.getPage() != null) {
94              params.add("page=" + documentViewer.getPage());
95          }
96  
97          // zoom level. Example: zoom=200 (accepted formats: '[zoom],[left],[top]',
98          // 'page-width', 'page-height', 'page-fit', 'auto')
99          if (!LangUtils.isValueBlank(documentViewer.getZoom())) {
100             params.add("zoom=" + documentViewer.getZoom());
101         }
102 
103         // nameddest: go to a named destination
104         if (!LangUtils.isValueBlank(documentViewer.getNameddest())) {
105             params.add("nameddest=" + documentViewer.getNameddest());
106         }
107 
108         // pagemode: either "thumbs" or "bookmarks". Example: pagemode=thumbs
109         if (!LangUtils.isValueBlank(documentViewer.getPagemode())) {
110             params.add("pagemode=" + documentViewer.getPagemode());
111         }
112 
113         if (!params.isEmpty()) {
114             return "#" + StringUtils.join(params, "&");
115         }
116         else {
117             return Constants.EMPTY_STRING;
118         }
119     }
120 
121     private String getResourceURL(final FacesContext context) {
122         final ResourceHandler handler = context.getApplication().getResourceHandler();
123         return context.getExternalContext().encodeResourceURL(
124                     handler.createResource("documentviewer/pdfviewer.html", "primefaces-extensions").getRequestPath());
125     }
126 
127     protected String getDocumentSource(final FacesContext context, final DocumentViewer documentViewer)
128                 throws UnsupportedEncodingException {
129         final String name = documentViewer.getName();
130 
131         if (name != null) {
132             final String libName = documentViewer.getLibrary();
133             final ResourceHandler handler = context.getApplication().getResourceHandler();
134             final Resource res = handler.createResource(name, libName);
135 
136             if (res == null) {
137                 return "RES_NOT_FOUND";
138             }
139             else {
140                 final String requestPath = res.getRequestPath();
141                 return context.getExternalContext().encodeResourceURL(requestPath);
142             }
143         }
144         else {
145             return DynamicContentSrcBuilder.build(context, documentViewer.getValue(), documentViewer,
146                         documentViewer.isCache(), DynamicContentType.STREAMED_CONTENT, true) + "&download="
147                         + documentViewer.getDownload();
148         }
149     }
