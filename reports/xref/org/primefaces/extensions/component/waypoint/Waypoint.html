156    /**
157     * {@inheritDoc}
158     */
159    @Override
160    public void queueEvent(final FacesEvent event) {
161       final FacesContext fc = FacesContext.getCurrentInstance();
162 
163       if (isSelfRequest(fc)) {
164          final Map<String, String> params = fc.getExternalContext().getRequestParameterMap();
165          final String eventName = params.get(Constants.RequestParams.PARTIAL_BEHAVIOR_EVENT_PARAM);
166          final AjaxBehaviorEvent behaviorEvent = (AjaxBehaviorEvent) event;
167 
168          if (WaypointEvent.NAME.equals(eventName)) {
169             final String direction = params.get(this.getClientId(fc) + "_direction");
170             final String waypointId = params.get(this.getClientId(fc) + "_waypointId");
171 
172             final WaypointEvent waypointEvent = new WaypointEvent(this, behaviorEvent.getBehavior(),
173                      direction != null ? WaypointEvent.Direction.valueOf(direction.toUpperCase(
174                               Locale.ENGLISH))
175                               : null,
176                      waypointId);
177             waypointEvent.setPhaseId(behaviorEvent.getPhaseId());
178             super.queueEvent(waypointEvent);
179 
180             return;
181          }
182       }
183 
184       super.queueEvent(event);
185    }

251    private boolean isSelfRequest(final FacesContext fc) {
