16  package org.primefaces.extensions.component.ckeditor;
17  
18  import java.io.IOException;
19  import java.util.Map;
20  
21  import javax.faces.component.UIComponent;
22  import javax.faces.context.FacesContext;
23  import javax.faces.context.ResponseWriter;
24  import javax.faces.convert.Converter;
25  
26  import org.primefaces.renderkit.InputRenderer;
27  import org.primefaces.util.ComponentUtils;
28  import org.primefaces.util.HTML;
29  import org.primefaces.util.WidgetBuilder;
30  
31  /**
32   * Renderer for the {@link CKEditor} component.
33   *
34   * @author Thomas Andraschko / last modified by $Author$
35   * @version $Revision$
36   * @since 0.2
37   */
38  public class CKEditorRenderer extends InputRenderer {
39  
40      @Override
41      public void decode(final FacesContext context, final UIComponent component) {
42          final CKEditor ckEditor = (CKEditor) component;
43  
44          if (!shouldDecode(ckEditor)) {
45              return;
46          }
47  
48          // set value
49          final String clientId = ckEditor.getClientId(context);
50          final Map<String, String> params = context.getExternalContext().getRequestParameterMap();
51          if (params.containsKey(clientId)) {
52              ckEditor.setSubmittedValue(params.get(clientId));
53          }
54  
55          // decode behaviors
56          decodeBehaviors(context, component);
57      }
58  
59      @Override
60      public void encodeEnd(final FacesContext context, final UIComponent component) throws IOException {
61          final CKEditor ckEditor = (CKEditor) component;
62  
63          encodeMarkup(context, ckEditor);
64          encodeScript(context, ckEditor);
65      }
66  
67      protected void encodeMarkup(final FacesContext context, final CKEditor ckEditor) throws IOException {
68          final ResponseWriter writer = context.getResponseWriter();
69          final String clientId = ckEditor.getClientId(context);
70  
71          writer.startElement("textarea", ckEditor);
72          writer.writeAttribute("id", clientId, null);
73          writer.writeAttribute("name", clientId, null);
74  
75          if (ckEditor.getTabindex() != null) {
76              writer.writeAttribute("tabindex", ckEditor.getTabindex(), null);
77          }
78  
79          renderAccessibilityAttributes(context, ckEditor);
80          renderPassThruAttributes(context, ckEditor, HTML.TEXTAREA_ATTRS_WITHOUT_EVENTS);
81          renderDomEvents(context, ckEditor, HTML.INPUT_TEXT_EVENTS);
82          renderValidationMetadata(context, ckEditor);
83  
84          final String valueToRender = ComponentUtils.getValueToRender(context, ckEditor);
85          if (valueToRender != null) {
86              if (ckEditor.isEscape()) {
87                  writer.writeText(valueToRender, null);
88              }
89              else {
90                  writer.write(valueToRender);
91              }
92          }
93  
94          writer.endElement("textarea");
95      }
96  
97      protected void encodeScript(final FacesContext context, final CKEditor ckEditor) throws IOException {
98          final WidgetBuilder wb = getWidgetBuilder(context);
99          wb.init("ExtCKEditor", ckEditor.resolveWidgetVar(), ckEditor.getClientId());
100         wb.attr("height", ckEditor.getHeight())
101                     .attr("width", ckEditor.getWidth())
102                     .attr("skin", ckEditor.getSkin())
103                     .attr("toolbar", ckEditor.getToolbar())
104                     .attr("readOnly", ckEditor.isReadonly())
105                     .attr("interfaceColor", ckEditor.getInterfaceColor())
106                     .attr("language", ckEditor.getLanguage())
107                     .attr("defaultLanguage", ckEditor.getDefaultLanguage());
108         if (ckEditor.getContentsCss() != null && ckEditor.getContentsCss().startsWith("[")) {
109             // new :: Array of CSS-Files :: ['/path/css1.css','/path/css2.css']
110             wb.nativeAttr("contentsCss", ckEditor.getContentsCss());
111         }
112         else {
113             // default behaviour
114             wb.attr("contentsCss", ckEditor.getContentsCss());
115         }
116         wb.attr("customConfig", ckEditor.getCustomConfig())
117                     .attr("advancedContentFilter", ckEditor.isAdvancedContentFilter())
118                     .attr("disableNativeSpellChecker", ckEditor.isDisableNativeSpellChecker())
119                     .attr("tabindex", ckEditor.getTabindex());
120 
121         encodeClientBehaviors(context, ckEditor);
122         wb.finish();
123     }
124 
125     @Override
126     public Object getConvertedValue(final FacesContext context, final UIComponent component,
127                 final Object submittedValue) {
128         final CKEditor ckEditor = (CKEditor) component;
129         final String value = (String) submittedValue;
130         final Converter converter = ComponentUtils.getConverter(context, component);
131 
132         if (converter != null) {
133             return converter.getAsObject(context, ckEditor, value);
134         }
135 
136         return value;
137     }
138 }
