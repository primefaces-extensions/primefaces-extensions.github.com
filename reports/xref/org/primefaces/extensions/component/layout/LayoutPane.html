334     @Override
335     public void processEvent(final ComponentSystemEvent event) throws AbortProcessingException {
336         super.processEvent(event);
337 
338         if (!(event instanceof PreRenderComponentEvent) || !getLayout().isBuildOptions()) {
339             return;
340         }
341 
342         setOptions(getParent());
343     }

345     public LayoutOptions getOptions() {
346         if (options != null) {
347             return options;
348         }
349 
350         // create options object
351         options = new LayoutOptions();
352 
353         final boolean isResizable = isResizable();
354         if (!isResizable) {
355             options.addOption(PropertyKeys.resizable.toString(), isResizable);
356         }
357 
358         final boolean isSlidable = isSlidable();
359         if (!isSlidable) {
360             options.addOption(PropertyKeys.slideable.toString(), isSlidable);
361         }
362 
363         final boolean isClosable = isClosable();
364         if (!isClosable) {
365             options.addOption(PropertyKeys.closable.toString(), isClosable);
366         }
367 
368         options.addOption(PropertyKeys.spacing_open.toString(), getSpacingOpen());
369         options.addOption(PropertyKeys.spacing_closed.toString(), getSpacingClosed());
370 
371         final boolean initClosed = isInitClosed();
372         if (initClosed) {
373             options.addOption(PropertyKeys.initClosed.toString(), initClosed);
374         }
375 
376         final boolean initHidden = isInitHidden();
377         if (initHidden) {
378             options.addOption(PropertyKeys.initHidden.toString(), initHidden);
379         }
380 
381         final boolean isResizeWhileDragging = isResizeWhileDragging();
382         if (isResizeWhileDragging) {
383             options.addOption(PropertyKeys.resizeWhileDragging.toString(), isResizeWhileDragging);
384         }
385 
386         final boolean isMaskContents = isMaskContents();
387         if (isMaskContents) {
388             options.addOption(PropertyKeys.maskContents.toString(), isMaskContents);
389             options.addOption("contentIgnoreSelector", ".ui-layout-mask");
390         }
391 
392         final boolean isMaskObjects = isMaskObjects();
393         if (isMaskObjects) {
394             options.addOption(PropertyKeys.maskObjects.toString(), isMaskObjects);
395             if (!isMaskContents) {
396                 options.addOption("contentIgnoreSelector", ".ui-layout-mask");
397             }
398         }
399 
400         final String paneSelector = getPaneSelector();
401         if (paneSelector != null) {
402             options.addOption(PropertyKeys.paneSelector.toString(), paneSelector);
403         }
404 
405         final String size = getSize();
406         if (size != null) {
407             options.addOption(PropertyKeys.size.toString(), size);
408         }
409 
410         final String minSize = getMinSize();
411         if (minSize != null) {
412             options.addOption(PropertyKeys.minSize.toString(), minSize);
413         }
414 
415         final String maxSize = getMaxSize();
416         if (maxSize != null) {
417             options.addOption(PropertyKeys.maxSize.toString(), maxSize);
418         }
419 
420         final String minWidth = getMinWidth();
421         if (minWidth != null) {
422             options.addOption(PropertyKeys.minWidth.toString(), minWidth);
423         }
424 
425         final String maxWidth = getMaxWidth();
426         if (maxWidth != null) {
427             options.addOption(PropertyKeys.maxWidth.toString(), maxWidth);
428         }
429 
430         final String minHeight = getMinHeight();
431         if (minHeight != null) {
432             options.addOption(PropertyKeys.minHeight.toString(), minHeight);
433         }
434 
435         final String maxHeight = getMaxHeight();
436         if (maxHeight != null) {
437             options.addOption(PropertyKeys.maxHeight.toString(), maxHeight);
438         }
439 
440         final LayoutOptionsl/layout/LayoutOptions.html#LayoutOptions">LayoutOptions tips = new LayoutOptions();
441         options.setTips(tips);
442 
443         final String resizerTip = getResizerTip();
444         if (resizerTip != null) {
445             tips.addOption(PropertyKeys.resizerTip.toString(), resizerTip);
446         }
447 
448         final String sliderTip = getSliderTip();
449         if (sliderTip != null) {
450             tips.addOption(PropertyKeys.sliderTip.toString(), sliderTip);
451         }
452 
453         final String togglerTipOpen = getTogglerTipOpen();
454         if (togglerTipOpen != null) {
455             tips.addOption(PropertyKeys.togglerTip_open.toString(), togglerTipOpen);
456         }
457 
458         final String togglerTipClosed = getTogglerTipClosed();
459         if (togglerTipClosed != null) {
460             tips.addOption(PropertyKeys.togglerTip_closed.toString(), togglerTipClosed);
461         }
462 
463         return options;
464     }

466     private void setOptions(final UIComponent parent) {
467         // create layout options for this pane via attributes defined in
468         // pe:layoutPane
469         final String position = getPosition();
470         final LayoutOptions thisLayoutOptions = getOptions();
471         LayoutOptions options;
472 
473         if (parent instanceof LayoutPane) {
474             final LayoutOptions parentLayoutOptions = ((LayoutPane) parent).getOptions();
475             options = parentLayoutOptions.getChildOptions();
476             if (options == null) {
477                 options = new LayoutOptions();
478                 parentLayoutOptions.setChildOptions(options);
479             }
480         }
481         else if (parent instanceof Layout) {
482             options = (LayoutOptions) ((Layout) parent).getOptions();
483             if (options == null) {
484                 final Layout layout = (Layout) parent;
485                 options = new LayoutOptions();
486                 layout.setOptions(options);
487 
488                 // options for all panes
489                 final LayoutOptionsyout/LayoutOptions.html#LayoutOptions">LayoutOptions defaults = new LayoutOptions();
490                 options.setPanesOptions(defaults);
491                 final LayoutOptionsl/layout/LayoutOptions.html#LayoutOptions">LayoutOptions tips = new LayoutOptions();
492                 defaults.setTips(tips);
493 
494                 final String resizerTip = layout.getResizerTip();
495                 if (resizerTip != null) {
496                     tips.addOption(Layout.PropertyKeys.resizerTip.toString(), resizerTip);
497                 }
498 
499                 final String sliderTip = layout.getSliderTip();
500                 if (sliderTip != null) {
501                     tips.addOption(Layout.PropertyKeys.sliderTip.toString(), sliderTip);
502                 }
503 
504                 final String togglerTipOpen = layout.getTogglerTipOpen();
505                 if (togglerTipOpen != null) {
506                     tips.addOption(Layout.PropertyKeys.togglerTip_open.toString(), togglerTipOpen);
507                 }
508 
509                 final String togglerTipClosed = layout.getTogglerTipClosed();
510                 if (togglerTipClosed != null) {
511                     tips.addOption(Layout.PropertyKeys.togglerTip_closed.toString(), togglerTipClosed);
512                 }
513 
514                 final boolean maskPanesEarly = layout.isMaskPanesEarly();
515                 if (maskPanesEarly) {
516                     options.addOption(Layout.PropertyKeys.maskPanesEarly.toString(), maskPanesEarly);
517                 }
518             }
519         }
520         else if (parent instanceof UIForm) {
521             // layout pane can be within a h:form
522             setOptions(parent.getParent());
523 
524             return;
525         }
526         else if (parent instanceof HtmlPanelGroup
527                     && Layout.STYLE_CLASS_LAYOUT_CONTENT.equals(((HtmlPanelGroup) parent).getStyleClass())
528                     && "block".equals(((HtmlPanelGroup) parent).getLayout())) {
529             // layout pane can be within h:panelGroup representing a HTML div
530             setOptions(parent.getParent());
531 
532             return;
533         }
534         else if (parent instanceof OutputPanel
535                     && Layout.STYLE_CLASS_LAYOUT_CONTENT.equals(((OutputPanel) parent).getStyleClass())) {
536             // layout pane can be within p:outputPanel representing a HTML div
537             setOptions(parent.getParent());
538 
539             return;
540         }
541         else if (parent != null && parent.toString().contains(Layout.STYLE_CLASS_LAYOUT_CONTENT)) {
542             // plain div (UIInstructions) with class "ui-layout-content"
543             setOptions(parent.getParent());
544 
545             return;
546         }
547         else {
548             throw new FacesException(
549                         "LayoutPane can be only placed within another LayoutPane, Layout, UIForm or DIV with class 'ui-layout-content'");
550         }
551 
552         if (Layout.PANE_POSITION_CENTER.equals(position)) {
553             options.setCenterOptions(thisLayoutOptions);
554         }
555         else if (Layout.PANE_POSITION_NORTH.equals(position)) {
556             options.setNorthOptions(thisLayoutOptions);
557         }
558         else if (Layout.PANE_POSITION_SOUTH.equals(position)) {
559             options.setSouthOptions(thisLayoutOptions);
560         }
561         else if (Layout.PANE_POSITION_WEST.equals(position)) {
562             options.setWestOptions(thisLayoutOptions);
563         }
564         else if (Layout.PANE_POSITION_EAST.equals(position)) {
565             options.setEastOptions(thisLayoutOptions);
566         }
567         else {
568             throw new FacesException("Pane position " + position
569                         + " is invalid. Valid positions are 'center', 'north' 'south', 'west', 'east'");
570         }
571     }

573     private Layout getLayout() {
574         UIComponent parent = getParent();
575 
576         while (!(parent instanceof Layout)) {
577             parent = parent.getParent();
578         }
579 
580         return (Layout) parent;
581     }
