16  package org.primefaces.extensions.component.masterdetail;
17  
18  import javax.faces.component.UIComponent;
19  import javax.faces.view.facelets.ComponentConfig;
20  import javax.faces.view.facelets.ComponentHandler;
21  import javax.faces.view.facelets.FaceletContext;
22  import javax.faces.view.facelets.MetaRuleset;
23  
24  import org.primefaces.component.breadcrumb.BreadCrumb;
25  import org.primefaces.facelets.MethodRule;
26  import org.primefaces.model.menu.DefaultMenuModel;
27  
28  /**
29   * {@link ComponentHandler} for the <code>MasterDetail</code>.
30   *
31   * @author Oleg Varaksin / last modified by $Author$
32   * @version $Revision$
33   * @since 0.2
34   */
35  public class MasterDetailTagHandler extends ComponentHandler {
36  
37      private static final MethodRule SELECT_LEVEL_LISTENER = new MethodRule("selectLevelListener", int.class, new Class[] {SelectLevelEvent.class});
38  
39      public MasterDetailTagHandler(final ComponentConfig config) {
40          super(config);
41      }
42  
43      @Override
44      protected MetaRuleset createMetaRuleset(final Class type) {
45          final MetaRuleset metaRuleset = super.createMetaRuleset(type);
46          metaRuleset.addRule(SELECT_LEVEL_LISTENER);
47  
48          return metaRuleset;
49      }
50  
51      @Override
52      public void onComponentPopulated(final FaceletContext ctx, final UIComponent c, final UIComponent parent) {
53          if (!isNew(parent)) {
54              return;
55          }
56  
57          final MasterDetail../org/primefaces/extensions/component/masterdetail/MasterDetail.html#MasterDetail">MasterDetail masterDetail = (MasterDetail) c;
58  
59          if (!isBreadcrumbAvailable(masterDetail)) {
60              // create BreadCrumb programmatically
61              final BreadCrumb breadcrumb = (BreadCrumb) ctx.getFacesContext().getApplication().createComponent(BreadCrumb.COMPONENT_TYPE);
62              breadcrumb.setId(masterDetail.getId() + MasterDetail.BREADCRUMB_ID_PREFIX);
63  
64              // set empty model
65              breadcrumb.setModel(new DefaultMenuModel());
66  
67              // add it to the MasterDetail
68              masterDetail.getChildren().add(breadcrumb);
69          }
70      }
71  
72      private static boolean isBreadcrumbAvailable(final MasterDetail masterDetail) {
73          for (final UIComponent child : masterDetail.getChildren()) {
74              if (child instanceof BreadCrumb) {
75                  return true;
76              }
77          }
78  
79          return false;
80      }
81  }
