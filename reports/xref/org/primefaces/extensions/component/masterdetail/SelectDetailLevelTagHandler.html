42  /**
43   * {@link TagHandler} for the <code>SelectDetailLevel</code>.
44   *
45   * @author Oleg Varaksin / last modified by $Author$
46   * @version $Revision$
47   * @since 0.2
48   */
49  public class SelectDetailLevelTagHandler extends TagHandler {
50  
51      private final TagAttribute contextValue;
52      private final TagAttribute listener;
53      private final TagAttribute level;
54      private final TagAttribute step;
55      private final TagAttribute preserveInputs;
56      private final TagAttribute resetInputs;
57      private final TagAttribute event;
58  
59      public SelectDetailLevelTagHandler(final TagConfig config) {
60          super(config);
61          contextValue = getAttribute("contextValue");
62          listener = getAttribute("listener");
63          level = getAttribute("level");
64          step = getAttribute("step");
65          preserveInputs = getAttribute("preserveInputs");
66          resetInputs = getAttribute("resetInputs");
67          event = super.getAttribute("event");
68      }
69  
70      @Override
71      public void apply(final FaceletContext ctx, final UIComponent parent) throws IOException {
72          if (!isAjaxifiedComponent(parent)) {
73              throw new FacesException("SelectDetailLevel must be only placed inside an ajaxified component.");
74          }
75  
76          if (!ComponentHandler.isNew(parent)) {
77              return;
78          }
79  
80          // get value expression for contextValue attribute of this tag handler
81          ValueExpression contextValueVE = null;
82          if (contextValue != null) {
83              contextValueVE = contextValue.getValueExpression(ctx, Object.class);
84              if (contextValueVE != null) {
85                  // store valid value expression in the attributes map
86                  parent.getAttributes().put(MasterDetail.CONTEXT_VALUE_VALUE_EXPRESSION, contextValueVE);
87              }
88          }
89  
90          // get value expression for level attribute of this tag handler
91          ValueExpression selectedLevelVE = null;
92          if (level != null) {
93              selectedLevelVE = level.getValueExpression(ctx, int.class);
94              if (selectedLevelVE != null) {
95                  // store valid value expression in the attributes map
96                  parent.getAttributes().put(MasterDetail.SELECTED_LEVEL_VALUE_EXPRESSION, selectedLevelVE);
97              }
98          }
99  
100         // get value expression for step attribute of this tag handler
101         ValueExpression selectedStepVE = null;
102         if (step != null) {
103             selectedStepVE = step.getValueExpression(ctx, int.class);
104             if (selectedStepVE != null) {
105                 // store valid value expression in the attributes map
106                 parent.getAttributes().put(MasterDetail.SELECTED_STEP_VALUE_EXPRESSION, selectedStepVE);
107             }
108         }
109 
110         // get value expression for preserveInputs attribute of this tag handler
111         ValueExpression preserveInputsVE = null;
112         if (preserveInputs != null) {
113             preserveInputsVE = preserveInputs.getValueExpression(ctx, String.class);
114             if (preserveInputsVE != null) {
115                 // store valid value expression in the attributes map
116                 parent.getAttributes().put(MasterDetail.PRESERVE_INPUTS_VALUE_EXPRESSION, preserveInputsVE);
117             }
118         }
119 
120         // get value expression for resetInputs attribute of this tag handler
121         ValueExpression resetInputsVE = null;
122         if (resetInputs != null) {
123             resetInputsVE = resetInputs.getValueExpression(ctx, String.class);
124             if (resetInputsVE != null) {
125                 // store valid value expression in the attributes map
126                 parent.getAttributes().put(MasterDetail.RESET_INPUTS_VALUE_EXPRESSION, resetInputsVE);
127             }
128         }
129 
130         // register a ComponentSystemEventListener
131         parent.subscribeToEvent(PreRenderComponentEvent.class, new PreRenderSourceListener());
132 
133         // register an ActionListener / AjaxBehaviorListener
134         if (listener != null) {
135             final MethodExpression me = listener.getMethodExpression(ctx, Object.class, new Class[] {Object.class});
136 
137             if (parent instanceof ActionSource) {
138                 ((ActionSource) parent).addActionListener(new SelectDetailLevelListener(me));
139             }
140             else if (parent instanceof ClientBehaviorHolder) {
141                 // find attached f:ajax / p:ajax corresponding to supported events
142                 final Collection<List<ClientBehavior>> clientBehaviors = getClientBehaviors(ctx, event, (ClientBehaviorHolder) parent);
143                 if (clientBehaviors == null || clientBehaviors.isEmpty()) {
144                     return;
145                 }
146 
147                 for (final List<ClientBehavior> listBehaviors : clientBehaviors) {
148                     for (final ClientBehavior clientBehavior : listBehaviors) {
149                         if (clientBehavior instanceof org.primefaces.behavior.ajax.AjaxBehavior) {
150                             ((org.primefaces.behavior.ajax.AjaxBehavior) clientBehavior).addAjaxBehaviorListener(
151                                         new SelectDetailLevelListener(me));
152                         }
153                         else if (clientBehavior instanceof javax.faces.component.behavior.AjaxBehavior) {
154                             ((javax.faces.component.behavior.AjaxBehavior) clientBehavior).addAjaxBehaviorListener(
155                                         new SelectDetailLevelListener(me));
156                         }
157                     }
158                 }
159             }
160         }
161     }
162 
163     public static boolean isAjaxifiedComponent(final UIComponent component) {
164         // check for ajax source
165         if (component instanceof AjaxSource && ((AjaxSource) component).isAjaxified()) {
166             return true;
167         }
168 
169         if (component instanceof ClientBehaviorHolder) {
170             // check for attached f:ajax / p:ajax
171             final Collection<List<ClientBehavior>> behaviors = ((ClientBehaviorHolder) component).getClientBehaviors()
172                         .values();
173             if (!behaviors.isEmpty()) {
174                 for (final List<ClientBehavior> listBehaviors : behaviors) {
175                     for (final ClientBehavior clientBehavior : listBehaviors) {
176                         if (clientBehavior instanceof javax.faces.component.behavior.AjaxBehavior
177                                     || clientBehavior instanceof org.primefaces.behavior.ajax.AjaxBehavior) {
178                             return true;
179                         }
180                     }
181                 }
182             }
183         }
184 
185         return false;
186     }
187 
188     public static Collection<List<ClientBehavior>> getClientBehaviors(final FaceletContext context, final TagAttribute event,
189                 final ClientBehaviorHolder clientBehaviorHolder) {
190         final Map<String, List<ClientBehavior>> mapBehaviors = clientBehaviorHolder.getClientBehaviors();
191         if (mapBehaviors == null || mapBehaviors.isEmpty()) {
192             return null;
193         }
194 
195         final String events = event != null ? event.getValue(context) : null;
196         final String[] arrEvents = events != null ? events.split("[\\s,]+") : null;
197         if (arrEvents == null || arrEvents.length < 1) {
198             return mapBehaviors.values();
199         }
200 
201         final Collection<List<ClientBehavior>> behaviors = new ArrayList<List<ClientBehavior>>();
202 
203         for (final Entry<String, List<ClientBehavior>> entry : mapBehaviors.entrySet()) {
204             if (ArrayUtils.contains(arrEvents, entry.getKey())) {
205                 behaviors.add(entry.getValue());
206             }
207         }
208 
209         return behaviors;
210     }
