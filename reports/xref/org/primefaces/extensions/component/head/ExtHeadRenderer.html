16  package org.primefaces.extensions.component.head;
17  
18  import java.io.IOException;
19  
20  import javax.faces.component.UIComponent;
21  import javax.faces.context.FacesContext;
22  import javax.faces.context.ResponseWriter;
23  
24  /**
25   * Renderer for the {@link ExtHead} component which extends PF {@link org.primefaces.renderkit.HeadRenderer}.
26   * <p>
27   * Ordering of rendered resources:
28   * </p>
29   *
30   * <pre>
31     - first facet if defined
32     - Theme CSS
33     - JSF, PF, PF Extensions CSS resources
34     - middle facet if defined
35     - JSF, PF, PF Extensions JS resources
36     - title
37     - shortcut icon
38     - h:head content (encoded by super class at encodeChildren)
39     - last facet if defined
40   * </pre>
41   *
42   * @author Thomas Andraschko / last modified by $Author$
43   * @version $Revision$
44   * @since 0.2
45   */
46  public class ExtHeadRenderer extends org.primefaces.renderkit.HeadRenderer {
47  
48      @Override
49      public void encodeEnd(final FacesContext context, final UIComponent component) throws IOException {
50          final ResponseWriter writer = context.getResponseWriter();
51          final ExtHead/../../../../org/primefaces/extensions/component/head/ExtHead.html#ExtHead">ExtHead extHead = (ExtHead) component;
52  
53          // encode title and shortcut icon
54          encodeTitle(extHead, writer);
55          encodeShortcutIcon(extHead, writer);
56  
57          super.encodeEnd(context, component);
58      }
59  
60      private void encodeTitle(final ExtHead extHead, final ResponseWriter writer) throws IOException {
61          if (extHead.getTitle() != null) {
62              writer.startElement("title", null);
63              writer.writeText(extHead.getTitle(), "title");
64              writer.endElement("title");
65          }
66      }
67  
68      private void encodeShortcutIcon(final ExtHead extHead, final ResponseWriter writer) throws IOException {
69          if (extHead.getShortcutIcon() != null) {
70              writer.startElement("link", null);
71              writer.writeAttribute("rel", "shortcut icon", null);
72              writer.writeAttribute("href", extHead.getShortcutIcon(), null);
73              writer.endElement("link");
74          }
75      }
76  }
