189 	@Override
190 	public String getClientId(final FacesContext context) {
191 		if (clientId != null) {
192 			return clientId;
193 		}
194 
195 		String id = getId();
196 		if (id == null) {
197 			final UniqueIdVendor parentUniqueIdVendor = ComponentTraversalUtils.closestUniqueIdVendor(this);
198 
199 			if (parentUniqueIdVendor == null) {
200 				final UIViewRoot viewRoot = context.getViewRoot();
201 
202 				if (viewRoot != null) {
203 					id = viewRoot.createUniqueId(context, null);
204 				} else {
205 					throw new FacesException("Cannot create clientId for " + this.getClass().getCanonicalName());
206 				}
207 			} else {
208 				id = parentUniqueIdVendor.createUniqueId(context, null);
209 			}
210 
211 			setId(id);
212 		}
213 
214 		final UIComponent namingContainer = ComponentTraversalUtils.closestNamingContainer(this);
215 		if (namingContainer != null) {
216 			final String containerClientId = namingContainer.getContainerClientId(context);
217 
218 			if (containerClientId != null) {
219             clientId = idBuilder.append(containerClientId).append(UINamingContainer.getSeparatorChar(context))
220                      .append(id)
221                      .toString();
222             idBuilder.setLength(0);
223          } else {
224 				clientId = id;
225 			}
226 		} else {
227 			clientId = id;
228 		}
229 
230 		final Renderer renderer = getRenderer(context);
231 		if (renderer != null) {
232 			clientId = renderer.convertClientId(context, clientId);
233 		}
234 
235 		return clientId;
236 	}

245 	@Override
246 	public String getContainerClientId(final FacesContext context) {
247 		final String clientId = this.getClientId(context);
248 
249 		final KeyData data = getData();
250 		final String key = data != null ? data.getKey() : null;
251 
252 		if (key == null) {
253 			return clientId;
254       } else {
255          final String containerClientId = idBuilder.append(clientId).append(UINamingContainer.getSeparatorChar(context))
256                   .append(key).toString();
257          idBuilder.setLength(0);
258 
259          return containerClientId;
260       }

263 	@Override
264 	public void processDecodes(final FacesContext context) {
265 		if (!isRendered()) {
266 			return;
267 		}
268 
269 		pushComponentToEL(context, this);
270 		preDecode(context);
271 		processFacets(context, PhaseId.APPLY_REQUEST_VALUES, this);
272 		processChildren(context, PhaseId.APPLY_REQUEST_VALUES);
273 
274 		try {
275 			decode(context);
276 		} catch (final RuntimeException e) {
277 			context.renderResponse();
278 			throw e;
279 		} finally {
280 			popComponentFromEL(context);
281 		}
282 	}

284 	@Override
285 	public void processValidators(final FacesContext context) {
286 		if (!isRendered()) {
287 			return;
288 		}
289 
290 		pushComponentToEL(context, this);
291 
292 		final Application app = context.getApplication();
293 		app.publishEvent(context, PreValidateEvent.class, this);
294 
295 		processFacets(context, PhaseId.PROCESS_VALIDATIONS, this);
296 		processChildren(context, PhaseId.PROCESS_VALIDATIONS);
297 
298 		app.publishEvent(context, PostValidateEvent.class, this);
299 		popComponentFromEL(context);
300 	}

302 	@Override
303 	public void processUpdates(final FacesContext context) {
304 		if (!isRendered()) {
305 			return;
306 		}
307 
308 		pushComponentToEL(context, this);
309 		processFacets(context, PhaseId.UPDATE_MODEL_VALUES, this);
310 		processChildren(context, PhaseId.UPDATE_MODEL_VALUES);
311 		popComponentFromEL(context);
312 	}

314 	protected void preDecode(final FacesContext context) {
315 		final Map<String, SavedEditableValueState> saved = (Map<String, SavedEditableValueState>) getStateHelper()
316 				.get(PropertyKeys.saved);
317 		if (null == saved) {
318 			getStateHelper().remove(PropertyKeys.saved);
319 		} else if (!keepSaved(context)) {
320 			for (final SavedEditableValueState saveState : saved.values()) {
321 				saveState.reset();
322 			}
323 		}
324 	}

326 	private boolean keepSaved(final FacesContext context) {
327 		return contextHasErrorMessages(context) || isNestedWithinIterator();
328 	}
329 
330 	private boolean contextHasErrorMessages(final FacesContext context) {
331 		final FacesMessage.Severity sev = context.getMaximumSeverity();
332 		return sev != null && FacesMessage.SEVERITY_ERROR.compareTo(sev) >= 0;
333 	}

335 	protected Boolean isNestedWithinIterator() {
336 		if (isNested == null) {
337 			UIComponent parent = this;
338          while (null != (parent = parent.getParent())) {
339             if (parent instanceof javax.faces.component.UIData
340                      || parent.getClass().getName().endsWith("UIRepeat")
341                      || parent instanceof UITabPanel && ((UITabPanel) parent).isRepeating()) { 
342                isNested = Boolean.TRUE;
343 					break;
344 				}
345 			}
346 			if (isNested == null) {
347 				isNested = Boolean.FALSE;
348 			}
349 		}
350 		return isNested;
351 	}

358 	@Override
359 	public void broadcast(final FacesEvent event) throws AbortProcessingException {
360 		if (!(event instanceof EventDataWrapper)) {
361 			super.broadcast(event);
362 
363 			return;
364 		}
365 
366 		final FacesContext context = FacesContext.getCurrentInstance();
367 		final KeyData oldData = getData();
368 		final EventDataWrapper eventDataWrapper = (EventDataWrapper) event;
369 		final FacesEvent originalEvent = eventDataWrapper.getFacesEvent();
370 		final UIComponent originalSource = (UIComponent) originalEvent.getSource();
371 		setData(eventDataWrapper.getData());
372 
373 		UIComponent compositeParent = null;
374 		try {
375 			if (!UIComponent.isCompositeComponent(originalSource)) {
376 				compositeParent = getCompositeComponentParent(originalSource);
377 			}
378 
379 			if (compositeParent != null) {
380 				compositeParent.pushComponentToEL(context, null);
381 			}
382 
383 			originalSource.pushComponentToEL(context, null);
384 			originalSource.broadcast(originalEvent);
385 		} finally {
386 			originalSource.popComponentFromEL(context);
387 			if (compositeParent != null) {
388 				compositeParent.popComponentFromEL(context);
389 			}
390 		}
391 
392 		setData(oldData);
393 	}

395 	@Override
396 	public boolean visitTree(final VisitContext context, final VisitCallback callback) {
397 		if (!isVisitable(context)) {
398 			return false;
399 		}
400 
401 		final FacesContext fc = context.getFacesContext();
402 		final KeyData oldData = getData();
403 		resetData();
404 
405 		pushComponentToEL(fc, null);
406 
407 		try {
408 			final VisitResult result = context.invokeVisitCallback(this, callback);
409 
410 			if (result == VisitResult.COMPLETE) {
411 				return true;
412 			}
413 
414 			if (result == VisitResult.ACCEPT && !context.getSubtreeIdsToVisit(this).isEmpty()) {
415 				if (getFacetCount() > 0) {
416 					for (final UIComponent facet : getFacets().values()) {
417 						if (facet.visitTree(context, callback)) {
418 							return true;
419 						}
420 					}
421 				}
422 
423 				if (visitChildren(context, callback)) {
424 					return true;
425 				}
426 			}
427 		} finally {
428 			popComponentFromEL(fc);
429 			setData(oldData);
430 		}
431 
432 		return false;
433 	}

435    @Override
436    public boolean invokeOnComponent(final FacesContext context, final String clientId, final ContextCallback callback) {
437       final KeyData oldData = getData();
438       resetData();
439 
440 		try {
441 			if (clientId.equals(super.getClientId(context))) {
442 				pushComponentToEL(context, getCompositeComponentParent(this));
443 				callback.invokeContextCallback(context, this);
444 
445 				return true;
446 			}
447 
448 			if (getFacetCount() > 0) {
449 				for (final UIComponent c : getFacets().values()) {
450 					if (clientId.equals(c.getClientId(context))) {
451 						callback.invokeContextCallback(context, c);
452 
453 						return true;
454 					}
455 				}
456 			}
457 
458 			return invokeOnChildren(context, clientId, callback);
459 		} catch (final FacesException fe) {
460 			throw fe;
461 		} catch (final Exception e) {
462 			throw new FacesException(e);
463 		} finally {
464 			popComponentFromEL(context);
465 			setData(oldData);
466 		}
467 	}

469 	protected void processFacets(final FacesContext context, final PhaseId phaseId, final UIComponent component) {
470 		resetData();
471 
472 		if (component.getFacetCount() > 0) {
473 			for (final UIComponent facet : component.getFacets().values()) {
474 				if (phaseId == PhaseId.APPLY_REQUEST_VALUES) {
475 					facet.processDecodes(context);
476 				} else if (phaseId == PhaseId.PROCESS_VALIDATIONS) {
477 					facet.processValidators(context);
478 				} else if (phaseId == PhaseId.UPDATE_MODEL_VALUES) {
479 					facet.processUpdates(context);
480 				} else {
481 					throw new IllegalArgumentException();
482 				}
483 			}
484 		}
485 	}

489 		final Integer i = (Integer) getStateHelper().get(PropertyKeys.lastId);
490 		int lastId = i != null ? i : 0;
491 		getStateHelper().put(PropertyKeys.lastId, ++lastId);
492 
493 		return UIViewRoot.UNIQUE_ID_PREFIX + (seed == null ? lastId : seed);
494 	}

496 	protected void exposeVar() {
497 		final FacesContext fc = FacesContext.getCurrentInstance();
498 		final Map<String, Object> requestMap = fc.getExternalContext().getRequestMap();
499 
500 		final String var = getVar();
501 		if (var != null) {
502 			final KeyData keyData = getData();
503 			if (keyData == null) {
504 				requestMap.remove(var);
505 			} else {
506 				requestMap.put(var, keyData.getData());
507 			}
508 		}
509 
510 		final String varContainerId = getVarContainerId();
511 		if (varContainerId != null) {
512 			final String containerClientId = getContainerClientId(fc);
513 			if (containerClientId == null) {
514 				requestMap.remove(varContainerId);
515 			} else {
516 				requestMap.put(varContainerId, containerClientId);
517 			}
518 		}
519 	}

527 	protected void saveDescendantState(final FacesContext context, final UIComponent component) {
528 		// force id reset
529 		component.setId(component.getId());
530 
531 		@SuppressWarnings("unchecked")
532 		final Map<String, SavedEditableValueState> saved = (Map<String, SavedEditableValueState>) getStateHelper()
533 				.get(PropertyKeys.saved);
534 
535 		if (component instanceof EditableValueHolder) {
536 			final EditableValueHolder input = (EditableValueHolder) component;
537 			SavedEditableValueState state = null;
538 			final String clientId = component.getClientId(context);
539 
540 			if (saved == null) {
541 				state = new SavedEditableValueState();
542 				getStateHelper().put(PropertyKeys.saved, clientId, state);
543 			}
544 
545 			if (state == null) {
546 				state = saved.get(clientId);
547 
548 				if (state == null) {
549 					state = new SavedEditableValueState();
550 					getStateHelper().put(PropertyKeys.saved, clientId, state);
551 				}
552 			}
553 
554 			state.setValue(input.getLocalValue());
555 			state.setValid(input.isValid());
556 			state.setSubmittedValue(input.getSubmittedValue());
557 			state.setLocalValueSet(input.isLocalValueSet());
558 			state.setLabelValue(((UIComponent) input).getAttributes().get("label"));
559 		}
560 
561 		for (final UIComponent child : component.getChildren()) {
562 			saveDescendantState(context, child);
563 		}
564 
565 		if (component.getFacetCount() > 0) {
566 			for (final UIComponent facet : component.getFacets().values()) {
567 				saveDescendantState(context, facet);
568 			}
569 		}
570 	}
571 
572 	protected void restoreDescendantState() {
573 		for (final UIComponent child : getChildren()) {
574 			restoreDescendantState(FacesContext.getCurrentInstance(), child);
575 		}
576 	}
577 
578 	protected void restoreDescendantState(final FacesContext context, final UIComponent component) {
579 		// force id reset
580 		component.setId(component.getId());
581 
582 		@SuppressWarnings("unchecked")
583 		final Map<String, SavedEditableValueState> saved = (Map<String, SavedEditableValueState>) getStateHelper()
584 				.get(PropertyKeys.saved);
585 
586 		if (component instanceof EditableValueHolder) {
587 			final EditableValueHolder input = (EditableValueHolder) component;
588 			final String clientId = component.getClientId(context);
589 
590 			SavedEditableValueState state = saved.get(clientId);
591 			if (state == null) {
592 				state = new SavedEditableValueState();
593 			}
594 
595 			input.setValue(state.getValue());
596 			input.setValid(state.isValid());
597 			input.setSubmittedValue(state.getSubmittedValue());
598 			input.setLocalValueSet(state.isLocalValueSet());
599 			if (state.getLabelValue() != null) {
600 				((UIComponent) input).getAttributes().put("label", state.getLabelValue());
601 			}
602 		}
603 
604 		for (final UIComponent child : component.getChildren()) {
605 			restoreDescendantState(context, child);
606 		}
607 
608 		if (component.getFacetCount() > 0) {
609 			for (final UIComponent facet : component.getFacets().values()) {
610 				restoreDescendantState(context, facet);
611 			}
612 		}
613 	}
