16  package org.primefaces.extensions.component.timer;
17  
18  import javax.faces.view.facelets.ComponentConfig;
19  import javax.faces.view.facelets.ComponentHandler;
20  import javax.faces.view.facelets.MetaRule;
21  import javax.faces.view.facelets.MetaRuleset;
22  
23  import org.primefaces.facelets.MethodRule;
24  
25  /**
26   * Handler for {@link Timer}
27   */
28  public class TimerHandler extends ComponentHandler {
29  
30      private static final MetaRule LISTENER = new MethodRule("listener", null, new Class[0]);
31  
32      public TimerHandler(ComponentConfig config) {
33          super(config);
34      }
35  
36      protected MetaRuleset createMetaRuleset(Class type) {
37          MetaRuleset metaRuleset = super.createMetaRuleset(type);
38  
39          metaRuleset.addRule(LISTENER);
40  
41          return metaRuleset;
42      }
43  }
