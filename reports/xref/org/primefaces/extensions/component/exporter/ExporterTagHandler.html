23  import javax.faces.FacesException;

33  /**
34   * <code>Exporter</code> component.
35   *
36   * @author Sudheer Jonna / last modified by $Author$
37   * @since 0.7.0
38   */
39  public class ExporterTagHandler extends TagHandler {
40  
41      private final TagAttribute target;
42      private final TagAttribute type;
43      private final TagAttribute fileName;
44      private final TagAttribute tableTitle;
45      private final TagAttribute pageOnly;
46      private final TagAttribute selectionOnly;
47      private final TagAttribute preProcessor;
48      private final TagAttribute postProcessor;
49      private final TagAttribute encoding;
50      private final TagAttribute subTable;
51      private final TagAttribute facetBackground;
52      private final TagAttribute facetFontSize;
53      private final TagAttribute facetFontColor;
54      private final TagAttribute facetFontStyle;
55      private final TagAttribute fontName;
56      private final TagAttribute cellFontSize;
57      private final TagAttribute cellFontColor;
58      private final TagAttribute cellFontStyle;
59      private final TagAttribute datasetPadding;
60      private final TagAttribute orientation;
61      private final TagAttribute skipComponents;
62  
63      public ExporterTagHandler(TagConfig tagConfig) {
64          super(tagConfig);
65          this.target = getRequiredAttribute("target");
66          this.type = getRequiredAttribute("type");
67          this.fileName = getAttribute("fileName");
68          this.tableTitle = getAttribute("tableTitle");
69          this.pageOnly = getAttribute("pageOnly");
70          this.selectionOnly = getAttribute("selectionOnly");
71          this.encoding = getAttribute("encoding");
72          this.preProcessor = getAttribute("preProcessor");
73          this.postProcessor = getAttribute("postProcessor");
74          this.subTable = getAttribute("subTable");
75          this.facetBackground = getAttribute("facetBackground");
76          this.facetFontSize = getAttribute("facetFontSize");
77          this.facetFontColor = getAttribute("facetFontColor");
78          this.facetFontStyle = getAttribute("facetFontStyle");
79          this.fontName = getAttribute("fontName");
80          this.cellFontSize = getAttribute("cellFontSize");
81          this.cellFontColor = getAttribute("cellFontColor");
82          this.cellFontStyle = getAttribute("cellFontStyle");
83          this.datasetPadding = getAttribute("datasetPadding");
84          this.orientation = getAttribute("orientation");
85          this.skipComponents = getAttribute("skipComponents");
86      }
87  
88      public void apply(FaceletContext faceletContext, UIComponent parent) throws IOException, FacesException, FaceletException, ELException {
89          if (ComponentHandler.isNew(parent)) {
90              ValueExpression targetVE = target.getValueExpression(faceletContext, Object.class);
91              ValueExpression typeVE = type.getValueExpression(faceletContext, Object.class);
92              ValueExpression fileNameVE = null;
93              ValueExpression tableTitleVE = null;
94              ValueExpression pageOnlyVE = null;
95              ValueExpression selectionOnlyVE = null;
96              ValueExpression encodingVE = null;
97              MethodExpression preProcessorME = null;
98              MethodExpression postProcessorME = null;
99              ValueExpression subTableVE = null;
100             ValueExpression facetBackgroundVE = null;
101             ValueExpression facetFontSizeVE = null;
102             ValueExpression facetFontColorVE = null;
103             ValueExpression facetFontStyleVE = null;
104             ValueExpression fontNameVE = null;
105             ValueExpression cellFontSizeVE = null;
106             ValueExpression cellFontColorVE = null;
107             ValueExpression cellFontStyleVE = null;
108             ValueExpression datasetPaddingVE = null;
109             ValueExpression orientationVE = null;
110             ValueExpression skipComponentsVE = null;
111 
112             if (fileName != null) {
113                 fileNameVE = fileName.getValueExpression(faceletContext, Object.class);
114             }
115             if (tableTitle != null) {
116                 tableTitleVE = tableTitle.getValueExpression(faceletContext, Object.class);
117             }
118             if (encoding != null) {
119                 encodingVE = encoding.getValueExpression(faceletContext, Object.class);
120             }
121             if (pageOnly != null) {
122                 pageOnlyVE = pageOnly.getValueExpression(faceletContext, Object.class);
123             }
124             if (selectionOnly != null) {
125                 selectionOnlyVE = selectionOnly.getValueExpression(faceletContext, Object.class);
126             }
127             if (preProcessor != null) {
128                 preProcessorME = preProcessor.getMethodExpression(faceletContext, null, new Class[]{Object.class});
129             }
130             if (postProcessor != null) {
131                 postProcessorME = postProcessor.getMethodExpression(faceletContext, null, new Class[]{Object.class});
132             }
133             if (subTable != null) {
134                 subTableVE = subTable.getValueExpression(faceletContext, Object.class);
135             }
136             if (facetBackground != null) {
137                 facetBackgroundVE = facetBackground.getValueExpression(faceletContext, Object.class);
138             }
139             if (facetFontSize != null) {
140                 facetFontSizeVE = facetFontSize.getValueExpression(faceletContext, Object.class);
141             }
142             if (facetFontColor != null) {
143                 facetFontColorVE = facetFontColor.getValueExpression(faceletContext, Object.class);
144             }
145             if (facetFontStyle != null) {
146                 facetFontStyleVE = facetFontStyle.getValueExpression(faceletContext, Object.class);
147             }
148             if (fontName != null) {
149                 fontNameVE = fontName.getValueExpression(faceletContext, Object.class);
150             }
151             if (cellFontSize != null) {
152                 cellFontSizeVE = cellFontSize.getValueExpression(faceletContext, Object.class);
153             }
154             if (cellFontColor != null) {
155                 cellFontColorVE = cellFontColor.getValueExpression(faceletContext, Object.class);
156             }
157             if (cellFontStyle != null) {
158                 cellFontStyleVE = cellFontStyle.getValueExpression(faceletContext, Object.class);
159             }
160             if (datasetPadding != null) {
161                 datasetPaddingVE = datasetPadding.getValueExpression(faceletContext, Object.class);
162             }
163             if (orientation != null) {
164                 orientationVE = orientation.getValueExpression(faceletContext, Object.class);
165             }
166             if (skipComponents != null) {
167                 skipComponentsVE = skipComponents.getValueExpression(faceletContext, Object.class);
168              }
169 
170             ActionSource actionSource = (ActionSource) parent;
171             actionSource.addActionListener(new DataExporter(targetVE, typeVE, fileNameVE, tableTitleVE, pageOnlyVE, selectionOnlyVE, encodingVE, preProcessorME, postProcessorME, subTableVE, facetBackgroundVE, facetFontSizeVE, facetFontColorVE, facetFontStyleVE, fontNameVE, cellFontSizeVE, cellFontColorVE, cellFontStyleVE, datasetPaddingVE, orientationVE,skipComponentsVE));
172 
173         }
174     }
