38  /**
39   * Renderer for the {@link CodeMirror} component.
40   *
41   * @author Thomas Andraschko / last modified by $Author$
42   * @version $Revision$
43   * @since 0.3
44   */
45  public class CodeMirrorRenderer extends InputRenderer {
46  
47      @Override
48      public void decode(final FacesContext facesContext, final UIComponent component) {
49          final CodeMirror codeMirror = (CodeMirror) component;
50  
51          if (codeMirror.isReadonly()) {
52              return;
53          }
54  
55          // set value
56          final String clientId = codeMirror.getClientId(facesContext);
57          final Map<String, String> params = facesContext.getExternalContext().getRequestParameterMap();
58          if (params.containsKey(clientId)) {
59              codeMirror.setSubmittedValue(params.get(clientId));
60          }
61  
62          // decode behaviors
63          decodeBehaviors(facesContext, component);
64  
65          // complete event
66          final String token = params.get(clientId + "_token");
67          if (token != null) {
68              final String context = params.get(clientId + "_context");
69              final int line = Integer.parseInt(params.get(clientId + "_line"));
70              final int column = Integer.parseInt(params.get(clientId + "_column"));
71  
72              final CompleteEvent autoCompleteEvent = new CompleteEvent(
73                      codeMirror, token, context, line, column);
74              autoCompleteEvent.setPhaseId(PhaseId.APPLY_REQUEST_VALUES);
75              codeMirror.queueEvent(autoCompleteEvent);
76          }
77      }
78  
79      @Override
80      public void encodeEnd(final FacesContext context, final UIComponent component) throws IOException {
81          final CodeMirror codeMirror = (CodeMirror) component;
82  
83          final Map<String, String> params = context.getExternalContext().getRequestParameterMap();
84          final String token = params.get(codeMirror.getClientId(context) + "_token");
85  
86          if (token != null) {
87              encodeSuggestions(context, codeMirror, codeMirror.getSuggestions());
88          } else {
89              encodeMarkup(context, codeMirror);
90              encodeScript(context, codeMirror);
91          }
92      }
93  
94      protected void encodeMarkup(final FacesContext context, final CodeMirror codeMirror) throws IOException {
95          final ResponseWriter writer = context.getResponseWriter();
96          final String clientId = codeMirror.getClientId(context);
97  
98          writer.startElement("textarea", codeMirror);
99          writer.writeAttribute("id", clientId, null);
100         writer.writeAttribute("name", clientId, null);
101 
102         renderPassThruAttributes(context, codeMirror, HTML.INPUT_TEXTAREA_ATTRS);
103         renderDomEvents(context, codeMirror, HTML.INPUT_TEXT_EVENTS);
104         
105         final String valueToRender = ComponentUtils.getValueToRender(context, codeMirror);
106         if (valueToRender != null) {
107             if (codeMirror.isEscape()) {
108                 writer.writeText(valueToRender, null);
109             } else {
110                 writer.write(valueToRender);
111             }
112         }
113 
114         writer.endElement("textarea");
115     }
116 
117     protected void encodeScript(final FacesContext context, final CodeMirror codeMirror) throws IOException {
118         WidgetBuilder wb = RequestContext.getCurrentInstance().getWidgetBuilder();
119         wb.initWithDomReady("ExtCodeMirror", codeMirror.resolveWidgetVar(), codeMirror.getClientId());
120         wb.attr("theme", codeMirror.getTheme())
121                 .attr("mode", codeMirror.getMode())
122                 .attr("indentUnit", codeMirror.getIndentUnit())
123                 .attr("smartIndent", codeMirror.isSmartIndent())
124                 .attr("tabSize", codeMirror.getTabSize())
125                 .attr("indentWithTabs", codeMirror.isIndentWithTabs())
126                 .attr("electricChars", codeMirror.isElectricChars())
127                 .attr("keyMap", codeMirror.getKeyMap())
128                 .attr("lineWrapping", codeMirror.isLineWrapping())
129                 .attr("lineNumbers", codeMirror.isLineNumbers())
130                 .attr("firstLineNumber", codeMirror.getFirstLineNumber())
131                 .attr("gutter", codeMirror.isGutter())
132                 .attr("fixedGutter", codeMirror.isFixedGutter())
133                 .attr("readOnly", codeMirror.isReadonly())
134                 .attr("matchBrackets", codeMirror.isMatchBrackets())
135                 .attr("workTime", codeMirror.getWorkTime())
136                 .attr("workDelay", codeMirror.getWorkDelay())
137                 .attr("pollInterval", codeMirror.getPollInterval())
138                 .attr("tabindex", codeMirror.getTabindex())
139                 .attr("undoDepth", codeMirror.getUndoDepth());
140 
141         if (codeMirror.getExtraKeys() != null) {
142             wb.append(",extraKeys:" + codeMirror.getExtraKeys());
143         }
144         if (!codeMirror.isGlobal()) {
145             wb.attr("global", false);
146         }
147         if (codeMirror.isAsync()) {
148             wb.attr("async", true);
149         }
150         if (codeMirror.getProcess() != null) {
151             wb.attr("process", SearchExpressionFacade.resolveClientIds(context, codeMirror, codeMirror.getProcess()));
152         }
153         if (codeMirror.getOnstart() != null) {
154             wb.callback("onstart", "function(request)", codeMirror.getOnstart());
155         }
156         if (codeMirror.getOncomplete() != null) {
157             wb.callback("oncomplete", "function(xhr, status, args)", codeMirror.getOncomplete());
158         }
159         if (codeMirror.getOnsuccess() != null) {
160             wb.callback("onsuccess", "function(data, status, xhr)", codeMirror.getOnsuccess());
161         }
162         if (codeMirror.getOnerror() != null) {
163             wb.callback("onerror", "function(xhr, status, error)", codeMirror.getOnerror());
164         }
165 
166         encodeClientBehaviors(context, codeMirror);
167 
168         wb.finish();
169     }
170 
171     @Override
172     public Object getConvertedValue(final FacesContext context, final UIComponent component, final Object submittedValue) {
173         final CodeMirror codeMirror = (CodeMirror) component;
174         final String value = (String) submittedValue;
175         final Converter converter = ComponentUtils.getConverter(context, component);
176 
177         if (converter != null) {
178             return converter.getAsObject(context, codeMirror, value);
179         }
180 
181         return value;
182     }
183 
184     protected void encodeSuggestions(final FacesContext context, final CodeMirror codeMirror, final List<String> suggestions) throws IOException {
185         final ResponseWriter writer = context.getResponseWriter();
186 
187         writer.startElement("ul", codeMirror);
188 
189         for (int i = 0; i < suggestions.size(); i++) {
190             final String suggestion = suggestions.get(i);
191 
192             writer.startElement("li", null);
193 
194             if (codeMirror.isEscapeSuggestions()) {
195                 writer.writeText(suggestion, null);
196             } else {
197                 writer.write(suggestion);
198             }
199 
200             writer.endElement("li");
201         }
202 
203         writer.endElement("ul");
204     }
