22  package org.primefaces.extensions.component.parameters;
23  
24  import java.util.Arrays;
25  
26  import javax.faces.FacesException;
27  import javax.faces.component.UIComponent;
28  import javax.faces.view.facelets.FaceletContext;
29  import javax.faces.view.facelets.TagAttribute;
30  import javax.faces.view.facelets.TagConfig;
31  import javax.faces.view.facelets.TagHandler;
32  
33  /**
34   * {@link TagHandler} for the <code>MethodSignature</code> tag.
35   *
36   * @author Thomas Andraschko / last modified by $Author$
37   * @version $Revision$
38   * @since 0.5
39   */
40  public class MethodSignatureTagHandler extends TagHandler {
41  
42      public static final String PARAMETERS_TYPES_ATTRIBUTE_NAME = "METHOD_SIGNATURE_PARAMETER_TYPES";
43  
44      private final Class<?>[] parameterTypes;
45  
46      public MethodSignatureTagHandler(final TagConfig config) {
47          super(config);
48  
49          final TagAttribute parametersTag = getRequiredAttribute("parameters");
50          try {
51              parameterTypes = parseParameterTypes(parametersTag.getValue());
52          }
53          catch (final ClassNotFoundException e) {
54              throw new FacesException(e.getMessage(), e);
55          }
56      }
57  
58      @Override
59      public void apply(final FaceletContext ctx, final UIComponent parent) {
60          // store all parameter types to parent component
61          parent.getAttributes().put(PARAMETERS_TYPES_ATTRIBUTE_NAME, parameterTypes);
62      }
63  
64      public Class<?>[] getParameterTypes() {
65          if (parameterTypes != null) {
66              return Arrays.copyOf(parameterTypes, parameterTypes.length);
67          }
68          return new Class<?>[] {};
69      }
70  
71      private Class<?>[] parseParameterTypes(final String parameters) throws ClassNotFoundException {
72          final String[] splitParameters = parameters.split(",");
73          final Class<?>[] types = new Class<?>[splitParameters.length];
74  
75          for (int i = 0; i < splitParameters.length; i++) {
76              types[i] = Class.forName(splitParameters[i].trim());
77          }
78  
79          return types;
80      }
81  }
