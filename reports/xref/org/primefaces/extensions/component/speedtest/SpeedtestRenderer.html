16  package org.primefaces.extensions.component.speedtest;
17  
18  import java.io.IOException;
19  
20  import javax.faces.FacesException;
21  import javax.faces.component.UIComponent;
22  import javax.faces.context.FacesContext;
23  import javax.faces.context.ResponseWriter;
24  
25  import org.primefaces.renderkit.CoreRenderer;
26  import org.primefaces.util.ComponentTraversalUtils;
27  import org.primefaces.util.HTML;
28  import org.primefaces.util.WidgetBuilder;
29  
30  /**
31   * Renderer for the {@link Speedtest} component.
32   *
33   * @author ssibitz ssibitz@me.com
34   * @since 6.2
35   */
36  public class SpeedtestRenderer extends CoreRenderer {
37  
38      /**
39       * {@inheritDoc}
40       */
41      @Override
42      public void decode(final FacesContext context, final UIComponent component) {
43          decodeBehaviors(context, component);
44      }
45  
46      /**
47       * {@inheritDoc}
48       */
49      @Override
50      public void encodeEnd(final FacesContext context, final UIComponent component) throws IOException {
51          final Speedtest speedtest = (Speedtest) component;
52          encodeMarkup(context, speedtest);
53          encodeScript(context, speedtest);
54      }
55  
56      /**
57       * Create the HTML markup for the DOM.
58       */
59      private void encodeMarkup(final FacesContext context, final Speedtest speedtest) throws IOException {
60          final ResponseWriter writer = context.getResponseWriter();
61          final String clientId = speedtest.getClientId(context);
62          final String widgetVar = speedtest.resolveWidgetVar();
63          // Generate Speedtest:
64          writer.startElement("div", speedtest);
65          writer.writeAttribute("id", clientId + "SpeedTest", "id");
66          if (speedtest.getStyleClass() != null) {
67              writer.writeAttribute("class", speedtest.getStyleClass(), "styleClass");
68          }
69          if (speedtest.getStyle() != null) {
70              writer.writeAttribute("style", speedtest.getStyle(), "style");
71          }
72          writer.writeAttribute(HTML.WIDGET_VAR, widgetVar, null);
73          writer.startElement("div", speedtest);
74          writer.writeAttribute("class", "ui-g", null);
75          // Download Gauge
76          writer.startElement("div", speedtest);
77          writer.writeAttribute("class", "ui-g-3", null);
78          writer.writeAttribute("id", clientId + "ggdown", "id");
79          writer.endElement("div");
80          // Upload Gauge
81          writer.startElement("div", speedtest);
82          writer.writeAttribute("class", "ui-g-3", null);
83          writer.writeAttribute("id", clientId + "ggup", "id");
84          writer.endElement("div");
85          // Ping Gauge
86          writer.startElement("div", speedtest);
87          writer.writeAttribute("class", "ui-g-3", null);
88          writer.writeAttribute("id", clientId + "ggping", "id");
89          writer.endElement("div");
90          // Jitter Gauge
91          writer.startElement("div", speedtest);
92          writer.writeAttribute("class", "ui-g-3", null);
93          writer.writeAttribute("id", clientId + "ggjitter", "id");
94          writer.endElement("div");
95          // Row End
96          writer.endElement("div");
97          // Speedtest End
98          writer.endElement("div");
99      }
100 
101     /**
102      * Create the Javascript.
103      */
104     private void encodeScript(final FacesContext context, final Speedtest speedtest) throws IOException {
105         final String clientId = speedtest.getClientId(context);
106         final UIComponent form = ComponentTraversalUtils.closestForm(context, speedtest);
107         if (form == null) {
108             throw new FacesException("Speedtest:" + clientId + " needs to be enclosed in a form component");
109         }
110 
111         final WidgetBuilder wb = getWidgetBuilder(context);
112         wb.init("ExtSpeedtest", speedtest.resolveWidgetVar(), clientId);
113         wb.attr("idDown", clientId + "ggdown");
114         wb.attr("idUp", clientId + "ggup");
115         wb.attr("idPing", clientId + "ggping");
116         wb.attr("idJitter", clientId + "ggjitter");
117         wb.attr("captionPing", speedtest.getCaptionPing());
118         wb.attr("captionJitter", speedtest.getCaptionJitter());
119         wb.attr("captionDownload", speedtest.getCaptionDownload());
120         wb.attr("captionUpload", speedtest.getCaptionUpload());
121         wb.attr("colorPing", speedtest.getColorPing());
122         wb.attr("colorJitter", speedtest.getColorJitter());
123         wb.attr("colorDownload", speedtest.getColorDownload());
124         wb.attr("colorUpload", speedtest.getColorUpload());
125         wb.attr("file", speedtest.getFile());
126         encodeClientBehaviors(context, speedtest);
127         wb.finish();
128     }
129 }
