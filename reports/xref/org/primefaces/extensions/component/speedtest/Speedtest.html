118     private Double convertParam(final String clientId, final String paramName, final Map<String, String> params) {
119         Double res = 0.0;
120         try {
121             res = Double.valueOf(params.get(clientId + paramName));
122         }
123         catch (final Exception e) {
124             throw new FacesException("Speedtest: can not convert result value for '" + paramName + "'");
125         }
126         return res;
127     }
128 
129     /**
130      * {@inheritDoc}
131      */
132     @Override
133     public void queueEvent(final FacesEvent event) {
134         final FacesContext fc = FacesContext.getCurrentInstance();
135 
136         if (isSelfRequest(fc) && event instanceof AjaxBehaviorEvent) {
137             final Map<String, String> params = fc.getExternalContext().getRequestParameterMap();
138             final String eventName = params.get(Constants.RequestParams.PARTIAL_BEHAVIOR_EVENT_PARAM);
139             final String clientId = this.getClientId(fc);
140             final AjaxBehaviorEvent behaviorEvent = (AjaxBehaviorEvent) event;
141             if (SpeedTestEvent.NAME.equals(eventName)) {
142                 // Get parameters:
143                 final Double pingTimeMS = convertParam(clientId, "_PingTimeMS", params);
144                 final Double jitterTimeMS = convertParam(clientId, "_JitterTimeMS", params);
145                 final Double speedMbpsDownload = convertParam(clientId, "_SpeedMbpsDownload", params);
146                 final Double speedMbpsUpload = convertParam(clientId, "_SpeedMbpsUpload", params);
147                 final SpeedTestEventEvent.html#SpeedTestEvent">SpeedTestEvent speedtestEvent = new SpeedTestEvent(this, behaviorEvent.getBehavior(),
148                             pingTimeMS, jitterTimeMS, speedMbpsDownload, speedMbpsUpload);
149                 speedtestEvent.setPhaseId(event.getPhaseId());
150                 super.queueEvent(speedtestEvent);
151                 return;
152             }
153         }
154         super.queueEvent(event);
155     }
156 
157     private boolean isSelfRequest(final FacesContext context) {
158         return this.getClientId(context)
159                     .equals(context.getExternalContext().getRequestParameterMap().get(
160                                 Constants.RequestParams.PARTIAL_SOURCE_PARAM));
161     }
