16  package org.primefaces.extensions.component.gravatar;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  
21  import javax.faces.component.UIOutput;
22  
23  public class Gravatar extends UIOutput {
24  
25      public static final String COMPONENT_TYPE = "org.primefaces.extensions.component.Gravatar";
26      public static final String COMPONENT_FAMILY = "org.primefaces.extensions.component";
27      public static final String DEFAULT_RENDERER = "org.primefaces.extensions.component.GravatarRenderer";
28      static final List<String> NOT_FOUND_VALUES = new ArrayList<String>();
29  
30      static {
31          NOT_FOUND_VALUES.add("default");
32          NOT_FOUND_VALUES.add("mm");
33          NOT_FOUND_VALUES.add("identicon");
34          NOT_FOUND_VALUES.add("monsterid");
35          NOT_FOUND_VALUES.add("wavatar");
36          NOT_FOUND_VALUES.add("retro");
37          NOT_FOUND_VALUES.add("blank");
38      }
39  
40      protected enum PropertyKeys {
41          notFound, size, style, qrCode, secure;
42      }
43  
44      public Gravatar() {
45          setRendererType(DEFAULT_RENDERER);
46      }
47  
48      @Override
49      public String getFamily() {
50          return COMPONENT_FAMILY;
51      }
52  
53      public Integer getSize() {
54          return (Integer) this.getStateHelper().eval(PropertyKeys.size, null);
55      }
56  
57      public void setSize(Integer size) {
58          this.getStateHelper().put(PropertyKeys.size, size);
59      }
60  
61      public String getStyle() {
62          return (String) this.getStateHelper().eval(PropertyKeys.style, null);
63      }
64  
65      public void setStyle(String style) {
66          this.getStateHelper().put(PropertyKeys.style, style);
67      }
68  
69      public boolean isQrCode() {
70          return Boolean.TRUE.equals(this.getStateHelper().eval(PropertyKeys.qrCode, false));
71      }
72  
73      public void setQrCode(boolean qrCode) {
74          this.getStateHelper().put(PropertyKeys.qrCode, qrCode);
75      }
76  
77      public boolean isSecure() {
78          return Boolean.TRUE.equals(this.getStateHelper().eval(PropertyKeys.secure, true));
79      }
80  
81      public void setSecure(boolean secure) {
82          this.getStateHelper().put(PropertyKeys.secure, secure);
83      }
84  
85      public String getNotFound() {
86          return (String) this.getStateHelper().eval(PropertyKeys.notFound, null);
87      }
88  
89      public void setNotFound(String notFound) {
90          this.getStateHelper().put(PropertyKeys.notFound, notFound);
91      }
92  
93  }
