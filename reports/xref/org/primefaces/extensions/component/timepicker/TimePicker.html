361         return appropriateLocale;

381     @Override
382     public void queueEvent(final FacesEvent event) {
383         final FacesContext fc = FacesContext.getCurrentInstance();
384         final String eventName = fc.getExternalContext().getRequestParameterMap()
385                     .get(Constants.RequestParams.PARTIAL_BEHAVIOR_EVENT_PARAM);
386 
387         if (isSelfRequest(fc) && event instanceof AjaxBehaviorEvent) {
388             if (TimeSelectEvent.NAME.equals(eventName)) {
389                 customEvents.put(TimeSelectEvent.NAME, (AjaxBehaviorEvent) event);
390 
391                 return;
392             }
393             else if (BeforeShowEvent.NAME.equals(eventName)) {
394                 final BeforeShowEventvent.html#BeforeShowEvent">BeforeShowEvent beforeShowEvent = new BeforeShowEvent(this,
395                             ((AjaxBehaviorEvent) event).getBehavior());
396                 beforeShowEvent.setPhaseId(event.getPhaseId());
397                 super.queueEvent(beforeShowEvent);
398 
399                 return;
400             }
401             else if (CloseEvent.NAME.equals(eventName)) {
402                 final CloseEventloseEvent.html#CloseEvent">CloseEvent closeEvent = new CloseEvent(this, ((AjaxBehaviorEvent) event).getBehavior());
403                 closeEvent.setPhaseId(event.getPhaseId());
404                 super.queueEvent(closeEvent);
405 
406                 return;
407             }
408         }
409 
410         super.queueEvent(event);
411     }

413     @Override
414     public void validate(final FacesContext fc) {
415         super.validate(fc);
416 
417         if (isValid()) {
418             for (final Entry<String, AjaxBehaviorEvent> entry : customEvents.entrySet()) {
419                 final AjaxBehaviorEvent behaviorEvent = entry.getValue();
420                 final TimeSelectEventvent.html#TimeSelectEvent">TimeSelectEvent timeSelectEvent = new TimeSelectEvent(this, behaviorEvent.getBehavior(), getValue());
421 
422                 if (behaviorEvent.getPhaseId().equals(PhaseId.APPLY_REQUEST_VALUES)) {
423                     timeSelectEvent.setPhaseId(PhaseId.PROCESS_VALIDATIONS);
424                 }
425                 else {
426                     timeSelectEvent.setPhaseId(PhaseId.INVOKE_APPLICATION);
427                 }
428 
429                 super.queueEvent(timeSelectEvent);
430             }
431         }
