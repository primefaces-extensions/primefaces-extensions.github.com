377    @Override
378    public void queueEvent(final FacesEvent event) {
379       final FacesContext fc = FacesContext.getCurrentInstance();
380       final String eventName = fc.getExternalContext().getRequestParameterMap()
381                .get(Constants.RequestParams.PARTIAL_BEHAVIOR_EVENT_PARAM);
382 
383       if (isSelfRequest(fc) && event instanceof AjaxBehaviorEvent) {
384          if (TimeSelectEvent.NAME.equals(eventName)) {
385             customEvents.put(TimeSelectEvent.NAME, (AjaxBehaviorEvent) event);
386 
387             return;
388          } else if (BeforeShowEvent.NAME.equals(eventName)) {
389             final BeforeShowEvent beforeShowEvent = new BeforeShowEvent(this,
390                      ((AjaxBehaviorEvent) event).getBehavior());
391             beforeShowEvent.setPhaseId(event.getPhaseId());
392             super.queueEvent(beforeShowEvent);
393 
394             return;
395          } else if (CloseEvent.NAME.equals(eventName)) {
396             final CloseEvent closeEvent = new CloseEvent(this, ((AjaxBehaviorEvent) event).getBehavior());
397             closeEvent.setPhaseId(event.getPhaseId());
398             super.queueEvent(closeEvent);
399 
400             return;
401          }
402       }
403 
404       super.queueEvent(event);
405    }
406 
407    @Override
408    public void validate(final FacesContext fc) {
409       super.validate(fc);
410 
411       if (isValid()) {
412          for (final Entry<String, AjaxBehaviorEvent> entry : customEvents.entrySet()) {
413             final AjaxBehaviorEvent behaviorEvent = entry.getValue();
414             final TimeSelectEvent timeSelectEvent = new TimeSelectEvent(this, behaviorEvent.getBehavior(),
415                      (Date) getValue());
416 
417             if (behaviorEvent.getPhaseId().equals(PhaseId.APPLY_REQUEST_VALUES)) {
418                timeSelectEvent.setPhaseId(PhaseId.PROCESS_VALIDATIONS);
419             } else {
420                timeSelectEvent.setPhaseId(PhaseId.INVOKE_APPLICATION);
421             }
422 
423             super.queueEvent(timeSelectEvent);
424          }
425       }
426    }
