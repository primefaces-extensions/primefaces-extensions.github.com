53      @Override
54      public Object getConvertedValue(FacesContext context, UIComponent component, Object submittedValue)
55                  throws ConverterException {
56          // only convert the values of the maps
57          if (submittedValue instanceof Map) {
58              final Map<String, Object> mapSub = (Map<String, Object>) submittedValue;
59              final List<String> keyValues = new ArrayList<>(mapSub.keySet());
60              final Map<String, Object> mapSubConv = new LinkedHashMap<>();
61              final TriStateManyCheckbox checkbox = (TriStateManyCheckbox) component;
62              final Converter converter = checkbox.getConverter();
63              if (converter != null) {
64                  for (final String keyVal : keyValues) {
65                      final Object mapVal = converter.getAsObject(context, checkbox, (String) mapSub.get(keyVal));
66                      mapSubConv.put(keyVal, mapVal);
67                  }
68  
69                  return mapSubConv;
70              }
71              else {
72                  return mapSub;
73              }
74          }
75          else {
76              throw new FacesException("Value of '" + component.getClientId() + "'must be a Map instance");
77          }
78      }

80      @Override
81      public void decode(FacesContext context, UIComponent component) {
82          final TriStateManyCheckbox checkbox = (TriStateManyCheckbox) component;
83          if (!shouldDecode(checkbox)) {
84              return;
85          }
86  
87          decodeBehaviors(context, checkbox);
88  
89          final String submitParam = getSubmitParam(context, checkbox);
90          final Map<String, String[]> params = context.getExternalContext().getRequestParameterValuesMap();
91  
92          String[] valuesArray = null;
93          if (params.containsKey(submitParam)) {
94              valuesArray = params.get(submitParam);
95          }
96  
97          checkbox.setSubmittedValue(getSubmitedMap(context, checkbox, valuesArray));
98      }

107     protected void encodeMarkup(FacesContext context, TriStateManyCheckbox checkbox) throws IOException {
108         final ResponseWriter writer = context.getResponseWriter();
109         final String clientId = checkbox.getClientId(context);
110         final String style = checkbox.getStyle();
111         String styleClass = checkbox.getStyleClass();
112         styleClass = styleClass == null ? SelectManyCheckbox.STYLE_CLASS : SelectManyCheckbox.STYLE_CLASS + " " + styleClass;
113 
114         writer.startElement("table", checkbox);
115         writer.writeAttribute("id", clientId, "id");
116         writer.writeAttribute("class", styleClass, "styleClass");
117         if (style != null) {
118             writer.writeAttribute("style", style, "style");
119         }
120 
121         encodeSelectItems(context, checkbox);
122 
123         writer.endElement("table");
124     }

126     protected void encodeSelectItems(FacesContext context, TriStateManyCheckbox checkbox) throws IOException {
127         final ResponseWriter writer = context.getResponseWriter();
128         final List<SelectItem> selectItems = getSelectItems(context, checkbox);
129         final Converter converter = checkbox.getConverter();
130         Map<String, Object> values = getValues(checkbox);
131         final Map<String, Object> submittedMap = getSubmittedFromComp(checkbox);
132         final String layout = checkbox.getLayout();
133         final boolean pageDirection = layout != null && layout.equals("pageDirection");
134 
135         if (submittedMap != null) {
136             values = submittedMap;
137         }
138 
139         if (converter != null && submittedMap == null) {
140             for (final Entry<String, Object> entry : values.entrySet()) {
141                 final String keyValue = converter.getAsString(context, checkbox, entry.getValue());
142                 values.put(entry.getKey(), keyValue);
143             }
144         }
145 
146         int idx = -1;
147         for (final SelectItem selectItem : selectItems) {
148             idx++;
149             if (pageDirection) {
150                 writer.startElement("tr", null);
151             }
152 
153             encodeOption(context, checkbox, values, converter, selectItem, idx);
154             if (pageDirection) {
155                 writer.endElement("tr");
156             }
157         }
158     }

160     protected void encodeOption(FacesContext context, UIInput component, Map<String, Object> values, Converter converter,
161                 SelectItem option, int idx) throws IOException {
162         final ResponseWriter writer = context.getResponseWriter();
163         final TriStateManyCheckbox checkbox = (TriStateManyCheckbox) component;
164         final String itemValueAsString = String.valueOf(option.getValue());
165         final String name = checkbox.getClientId(context);
166         final String id = name + UINamingContainer.getSeparatorChar(context) + idx;
167         final boolean disabled = option.isDisabled() || checkbox.isDisabled();
168 
169         final String itemValue = (String) option.getValue();
170 
171         final int valueInput = getValueForInput(context, component, itemValue, values, converter);
172         if (option.isNoSelectionOption() && values != null && Constants.EMPTY_STRING.equals(itemValue)) {
173             return;
174         }
175 
176         writer.startElement("td", null);
177 
178         writer.startElement("div", null);
179         writer.writeAttribute("class", HTML.CHECKBOX_CLASS, null);
180 
181         encodeOptionInput(context, checkbox, id, name, disabled, itemValueAsString, valueInput);
182         encodeOptionOutput(context, checkbox, valueInput, disabled);
183 
184         writer.endElement("div");
185         writer.endElement("td");
186 
187         writer.startElement("td", null);
188         encodeOptionLabel(context, checkbox, id, option, disabled);
189         writer.endElement("td");
190     }

192     protected void encodeOptionInput(FacesContext context, TriStateManyCheckbox checkbox, String id, String name,
193                 boolean disabled, String value, int valueInput) throws IOException {
194         final ResponseWriter writer = context.getResponseWriter();
195 
196         writer.startElement("div", null);
197         writer.writeAttribute("class", HTML.CHECKBOX_INPUT_WRAPPER_CLASS, null);
198 
199         writer.startElement("input", null);
200         writer.writeAttribute("id", id, null);
201         writer.writeAttribute("name", name, null);
202         writer.writeAttribute("type", "text", null);
203         writer.writeAttribute("value", valueInput, null);
204         writer.writeAttribute("itemValue", value, null);
205 
206         if (disabled) {
207             writer.writeAttribute("disabled", "disabled", null);
208         }
209 
210         if (checkbox.getOnchange() != null) {
211             writer.writeAttribute("onchange", checkbox.getOnchange(), null);
212         }
213 
214         writer.endElement("input");
215 
216         writer.endElement("div");
217     }

219     protected void encodeOptionOutput(final FacesContext context, final TriStateManyCheckbox checkbox, final int valCheck,
220                 final boolean disabled) throws IOException {
221         final ResponseWriter writer = context.getResponseWriter();
222         String styleClass = HTML.CHECKBOX_BOX_CLASS;
223         styleClass = valCheck == 1 || valCheck == 2 ? styleClass + " ui-state-active" : styleClass;
224         styleClass = disabled ? styleClass + " ui-state-disabled" : styleClass;
225 
226         // if stateIcon is defined use it insted of default icons.
227         final String stateOneIconClass = checkbox.getStateOneIcon() != null ? TriStateManyCheckbox.UI_ICON + checkbox.getStateOneIcon()
228                     : Constants.EMPTY_STRING;
229         final String stateTwoIconClass = checkbox.getStateTwoIcon() != null ? TriStateManyCheckbox.UI_ICON + checkbox.getStateTwoIcon()
230                     : TriStateManyCheckbox.UI_ICON + "ui-icon-check";
231         final String stataThreeIconClass = checkbox.getStateThreeIcon() != null ? TriStateManyCheckbox.UI_ICON + checkbox.getStateThreeIcon()
232                     : TriStateManyCheckbox.UI_ICON + "ui-icon-closethick";
233 
234         final String statesIconsClasses = "[\"" + stateOneIconClass + "\",\"" + stateTwoIconClass + "\",\"" + stataThreeIconClass + "\"]";
235 
236         final String statesTitles = "[\"" + EscapeUtils.forJavaScript(checkbox.getStateOneTitle()) + "\",\""
237                     + EscapeUtils.forJavaScript(checkbox.getStateTwoTitle()) + "\",\"" + EscapeUtils.forJavaScript(checkbox.getStateThreeTitle()) + "\"]";
238 
239         String iconClass = "ui-chkbox-icon ui-c"; // HTML.CHECKBOX_ICON_CLASS;
240         String activeTitle = Constants.EMPTY_STRING;
241         if (valCheck == 0) {
242             iconClass = iconClass + " " + stateOneIconClass;
243             activeTitle = checkbox.getStateOneTitle();
244         }
245         else if (valCheck == 1) {
246             iconClass = iconClass + " " + stateTwoIconClass;
247             activeTitle = checkbox.getStateTwoTitle();
248         }
249         else if (valCheck == 2) {
250             iconClass = iconClass + " " + stataThreeIconClass;
251             activeTitle = checkbox.getStateThreeTitle();
252         }
253 
254         String dataTitles = Constants.EMPTY_STRING;
255         String titleAtt = Constants.EMPTY_STRING;
256 
257         if (!checkbox.getStateOneTitle().isEmpty()
258                     || !checkbox.getStateTwoTitle().isEmpty() || !checkbox.getStateThreeTitle().isEmpty()) {
259             // preparation with singe quotes for .data('titlestates')
260             dataTitles = "data-titlestates='" + statesTitles + "' ";
261             // active title Att
262             titleAtt = " title=\"" + EscapeUtils.forJavaScript(activeTitle) + "\" ";
263         }
264 
265         String tabIndexTag = " tabIndex=0 ";
266         if (checkbox.getTabindex() != null) {
267             tabIndexTag = "tabIndex=" + checkbox.getTabindex() + " ";
268         }
269 
270         // preparation with singe quotes for .data('iconstates')
271         writer.write("<div " + tabIndexTag + titleAtt + "class=\"" + styleClass + "\" data-iconstates='" + statesIconsClasses + "' "
272                     + dataTitles + ">"
273                     + "<span class=\"" + iconClass + "\"></span></div>");
274 
275     }

277     protected void encodeScript(FacesContext context, TriStateManyCheckbox checkbox) throws IOException {
278         final WidgetBuilder wb = getWidgetBuilder(context);
279         wb.init("ExtTriStateManyCheckbox", checkbox.resolveWidgetVar(), checkbox.getClientId());
280         encodeClientBehaviors(context, checkbox);
281         wb.finish();
282     }

284     protected void encodeOptionLabel(FacesContext context, TriStateManyCheckbox checkbox, String containerClientId,
285                 SelectItem option, boolean disabled) throws IOException {
286         final ResponseWriter writer = context.getResponseWriter();
287 
288         writer.startElement("label", null);
289         writer.writeAttribute("for", containerClientId, null);
290         if (disabled) {
291             writer.writeAttribute("class", "ui-state-disabled", null);
292         }
293 
294         if (option.isEscape()) {
295             writer.writeText(option.getLabel(), null);
296         }
297         else {
298             writer.write(option.getLabel());
299         }
300 
301         writer.endElement("label");
302     }

309     /*
310      * return the value for the triState based on the value of the selectItems on the iteration
311      */
312     protected int getValueForInput(FacesContext context, UIInput component, String itemValue, Map<String, Object> valueArray,
313                 Converter converter) {
314         try {
315             int retInt = 0;
316             if (itemValue == null || valueArray == null) {
317                 return retInt;
318             }
319 
320             if (valueArray.containsKey(itemValue)) {
321                 retInt = Integer.parseInt((String) valueArray.get(itemValue));
322 
323                 return retInt % 3;
324             }
325             else {
326                 return retInt;
327             }
328         }
329         catch (final NumberFormatException ex) {
330             throw new FacesException("State of '" + component.getClientId() + "' must be an integer representation");
331         }
332     }

334     @Override
335     protected Map<String, Object> getValues(UIComponent component) {
336         final UISelectMany selectMany = (UISelectMany) component;
337         final Object value = selectMany.getValue();
338 
339         if (value == null) {
340             return null;
341         }
342         else if (value instanceof Map) {
343             // it should be a Map instance for <ItemStringValue,Value>
344             return (Map) value;
345         }
346         else {
347             throw new FacesException("Value of '" + component.getClientId() + "'must be a Map instance");
348         }
349     }

351     protected Map<String, Object> getSubmitedMap(FacesContext context, TriStateManyCheckbox checkbox, String[] valuesArray) {
352         final List<SelectItem> selectItems = getSelectItems(context, checkbox);
353         final Map<String, Object> subValues = new LinkedHashMap<>();
354         if (valuesArray != null && valuesArray.length == selectItems.size()) {
355             int idx = -1;
356             for (final SelectItem item : selectItems) {
357                 idx++;
358 
359                 final String keyMap = (String) item.getValue();
360                 final Object valueMap = valuesArray[idx];
361                 subValues.put(keyMap, valueMap);
362             }
363 
364             return subValues;
365         }
366         else {
367             return null;
368         }
369     }

371     protected Map<String, Object> getSubmittedFromComp(UIComponent component) {
372         final TriStateManyCheckbox checkbox = (TriStateManyCheckbox) component;
373         final Map<String, Object> ret = (Map<String, Object>) checkbox.getSubmittedValue();
374         if (ret != null) {
375             final Map<String, Object> subValues = new LinkedHashMap<>();
376 
377             // need to reverse the order of element on the map to take the value as on decode.
378             final Set<String> keys = ret.keySet();
379             final String[] tempArray = keys.toArray(new String[keys.size()]);
380 
381             final int length = tempArray.length;
382             for (int i = length - 1; i >= 0; i--) {
383                 final String key = tempArray[i];
384                 final Object val = ret.get(key);
385                 subValues.put(key, val);
386             }
387 
388             return subValues;
389         }
390         else {
391             return null;
392         }
