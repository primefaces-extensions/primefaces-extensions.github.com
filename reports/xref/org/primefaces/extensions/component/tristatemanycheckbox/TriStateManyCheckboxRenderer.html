54  	@Override
55  	public Object getConvertedValue(FacesContext context, UIComponent component, Object submittedValue)
56  	    throws ConverterException {
57  		//only convert the values of the maps
58  		if (submittedValue instanceof Map) {
59  			@SuppressWarnings("unchecked")
60  			Map<String, Object> mapSub = (Map<String, Object>) submittedValue;
61  			List<String> keyValues = new ArrayList<String>(mapSub.keySet());
62  			Map<String, Object> mapSubConv = new LinkedHashMap<String, Object>();
63  			TriStateManyCheckbox checkbox = (TriStateManyCheckbox) component;
64  			Converter converter = checkbox.getConverter();
65  			if (converter != null) {
66  				for (String keyVal : keyValues) {
67  					Object mapVal = converter.getAsObject(context, checkbox, (String) mapSub.get(keyVal));
68  					mapSubConv.put(keyVal, mapVal);
69  				}
70  
71  				return mapSubConv;
72  			} else {
73  				return mapSub;
74  			}
75  		} else {
76  			throw new FacesException("Value of '" + component.getClientId() + "'must be a Map instance");
77  		}
78  	}

80  	@Override
81  	public void decode(FacesContext context, UIComponent component) {
82  		if (!shouldDecode(component)) {
83  			return;
84  		}
85  
86  		TriStateManyCheckbox checkbox = (TriStateManyCheckbox) component;
87  		decodeBehaviors(context, checkbox);
88  
89  		String submitParam = getSubmitParam(context, checkbox);
90  		Map<String, String[]> params = context.getExternalContext().getRequestParameterValuesMap();
91  
92  		String[] valuesArray = null;
93  		if (params.containsKey(submitParam)) {
94  			valuesArray = params.get(submitParam);
95  		}
96  
97  		checkbox.setSubmittedValue(getSubmitedMap(context, checkbox, valuesArray));
98  	}

107 	protected void encodeMarkup(FacesContext context, TriStateManyCheckbox checkbox) throws IOException {
108 		ResponseWriter writer = context.getResponseWriter();
109 		String clientId = checkbox.getClientId(context);
110 		String style = checkbox.getStyle();
111 		String styleClass = checkbox.getStyleClass();
112 		styleClass = styleClass == null ? SelectManyCheckbox.STYLE_CLASS : SelectManyCheckbox.STYLE_CLASS + " " + styleClass;
113 
114 		writer.startElement("table", checkbox);
115 		writer.writeAttribute("id", clientId, "id");
116 		writer.writeAttribute("class", styleClass, "styleClass");
117 		if (style != null) {
118 			writer.writeAttribute("style", style, "style");
119 		}
120 
121 		encodeSelectItems(context, checkbox);
122 
123 		writer.endElement("table");
124 	}

126 	protected void encodeSelectItems(FacesContext context, TriStateManyCheckbox checkbox) throws IOException {
127 		ResponseWriter writer = context.getResponseWriter();
128 		List<SelectItem> selectItems = getSelectItems(context, checkbox);
129 		Converter converter = checkbox.getConverter();
130 		Map<String, Object> values = getValues(checkbox);
131 		Map<String, Object> submittedMap = getSubmittedFromComp(checkbox);
132 		String layout = checkbox.getLayout();
133 		boolean pageDirection = layout != null && layout.equals("pageDirection");
134 
135 		if (submittedMap != null) {
136 			values = submittedMap;
137 		}
138 
139 		if (converter != null && submittedMap == null) {
140 			for (Entry<String, Object> entry : values.entrySet()) {
141 				String keyValue = converter.getAsString(context, checkbox, entry.getValue());
142 				values.put(entry.getKey(), keyValue);
143 			}
144 		}
145 
146 		int idx = -1;
147 		for (SelectItem selectItem : selectItems) {
148 			idx++;
149 			if (pageDirection) {
150 				writer.startElement("tr", null);
151 			}
152 
153 			encodeOption(context, checkbox, values, converter, selectItem, idx);
154 			if (pageDirection) {
155 				writer.endElement("tr");
156 			}
157 		}
158 	}

160 	protected void encodeOption(FacesContext context, UIInput component, Map<String, Object> values, Converter converter,
161 	                            SelectItem option, int idx) throws IOException {
162 		ResponseWriter writer = context.getResponseWriter();
163 		TriStateManyCheckbox checkbox = (TriStateManyCheckbox) component;
164 		String itemValueAsString = String.valueOf(option.getValue());
165 		String name = checkbox.getClientId(context);
166 		String id = name + UINamingContainer.getSeparatorChar(context) + idx;
167 		boolean disabled = option.isDisabled() || checkbox.isDisabled();
168 
169 		String itemValue = (String) option.getValue();
170 
171 		int valueInput = getValueForInput(context, component, itemValue, values, converter);
172 		if (option.isNoSelectionOption() && values != null && "".equals(itemValue)) {
173 			return;
174 		}
175 
176 		writer.startElement("td", null);
177 
178 		writer.startElement("div", null);
179 		writer.writeAttribute("class", HTML.CHECKBOX_CLASS, null);
180 
181 		encodeOptionInput(context, checkbox, id, name, disabled, itemValueAsString, valueInput);
182 		encodeOptionOutput(context, checkbox, valueInput, disabled);
183 
184 		writer.endElement("div");
185 		writer.endElement("td");
186 
187 		writer.startElement("td", null);
188 		encodeOptionLabel(context, checkbox, id, option, disabled);
189 		writer.endElement("td");
190 	}

192 	protected void encodeOptionInput(FacesContext context, TriStateManyCheckbox checkbox, String id, String name,
193 	                                 boolean disabled, String value, int valueInput) throws IOException {
194 		ResponseWriter writer = context.getResponseWriter();
195 
196 		writer.startElement("div", null);
197 		writer.writeAttribute("class", HTML.CHECKBOX_INPUT_WRAPPER_CLASS, null);
198 
199 		writer.startElement("input", null);
200 		writer.writeAttribute("id", id, null);
201 		writer.writeAttribute("name", name, null);
202 		writer.writeAttribute("type", "text", null);
203 		writer.writeAttribute("value", valueInput, null);
204 		writer.writeAttribute("itemValue", value, null);
205 
206 		if (disabled) {
207 			writer.writeAttribute("disabled", "disabled", null);
208 		}
209 
210 		if (checkbox.getOnchange() != null) {
211 			writer.writeAttribute("onchange", checkbox.getOnchange(), null);
212 		}
213 
214 		writer.endElement("input");
215 
216 		writer.endElement("div");
217 	}

219 	protected void encodeOptionOutput(final FacesContext context, final TriStateManyCheckbox checkbox, final int valCheck,
220 	                                  final boolean disabled) throws IOException {
221 		ResponseWriter writer = context.getResponseWriter();
222 		String styleClass = HTML.CHECKBOX_BOX_CLASS;
223 		styleClass = (valCheck == 1 || valCheck == 2) ? styleClass + " ui-state-active" : styleClass; 
224 		styleClass = disabled ? styleClass + " ui-state-disabled" : styleClass;
225 
226 		//if stateIcon is defined use it insted of default icons.
227 		String stateOneIconClass =
228 		    checkbox.getStateOneIcon() != null ? TriStateManyCheckbox.UI_ICON + checkbox.getStateOneIcon() : "";
229 		String stateTwoIconClass =
230 		    checkbox.getStateTwoIcon() != null ? TriStateManyCheckbox.UI_ICON + checkbox.getStateTwoIcon()
231 		                                       : TriStateManyCheckbox.UI_ICON + "ui-icon-check";
232 		String stataThreeIconClass =
233 		    checkbox.getStateThreeIcon() != null ? TriStateManyCheckbox.UI_ICON + checkbox.getStateThreeIcon()
234 		                                         : TriStateManyCheckbox.UI_ICON + "ui-icon-closethick";
235 
236 		String statesIconsClasses =
237 		    "[\"" + stateOneIconClass + "\",\"" + stateTwoIconClass + "\",\"" + stataThreeIconClass + "\"]";
238                 
239                 String statesTitles =
240 		    "[\"" + checkbox.getStateOneTitle() + "\",\"" 
241                        + checkbox.getStateTwoTitle() + "\",\"" + checkbox.getStateThreeTitle() + "\"]";
242                 
243                 
244 
245 		String iconClass = "ui-chkbox-icon ui-c"; //HTML.CHECKBOX_ICON_CLASS;
246                 String activeTitle = "";
247 		if (valCheck == 0) {
248 			iconClass = iconClass + " " + stateOneIconClass;
249                         activeTitle = checkbox.getStateOneTitle();
250 		} else if (valCheck == 1) {
251 			iconClass = iconClass + " " + stateTwoIconClass;
252                         activeTitle = checkbox.getStateTwoTitle();
253 		} else if (valCheck == 2) {
254 			iconClass = iconClass + " " + stataThreeIconClass;
255                         activeTitle = checkbox.getStateThreeTitle();
256 		}
257                 
258                 String dataTitles = "";
259                 String titleAtt = "";
260                 
261                 if(!checkbox.getStateOneTitle().isEmpty() 
262                     || !checkbox.getStateTwoTitle().isEmpty() || !checkbox.getStateThreeTitle().isEmpty()){
263                         // preparation with singe quotes for .data('titlestates')
264                         dataTitles = "data-titlestates='" + statesTitles + "' ";
265                         // active title Att 
266                         titleAtt = " title=\"" + activeTitle + "\" ";
267                 }                
268 
269                 String tabIndexTag = " tabIndex=0 ";                
270                 if(checkbox.getTabindex()!=null){
271                     tabIndexTag = "tabIndex=" + checkbox.getTabindex()+ " ";
272                 }      
273                 
274                 // preparation with singe quotes for .data('iconstates') 
275 		writer.write("<div " + tabIndexTag + titleAtt + "class=\"" + styleClass + "\" data-iconstates='" + statesIconsClasses + "' "
276                                 + dataTitles + ">"
277                                 + "<span class=\"" + iconClass + "\"></span></div>");                
278 		
279 	}

281 	protected void encodeScript(FacesContext context, TriStateManyCheckbox checkbox) throws IOException {
282         WidgetBuilder wb = RequestContext.getCurrentInstance().getWidgetBuilder();
283         wb.initWithDomReady("ExtTriStateManyCheckbox", checkbox.resolveWidgetVar(), checkbox.getClientId());
284         encodeClientBehaviors(context, checkbox);
285         wb.finish();
286 	}

288 	protected void encodeOptionLabel(FacesContext context, TriStateManyCheckbox checkbox, String containerClientId,
289 	                                 SelectItem option, boolean disabled) throws IOException {
290 		ResponseWriter writer = context.getResponseWriter();
291 
292 		writer.startElement("label", null);
293 		writer.writeAttribute("for", containerClientId, null);
294 		if (disabled) {
295 			writer.writeAttribute("class", "ui-state-disabled", null);
296 		}
297 
298 		if (option.isEscape()) {
299 			writer.writeText(option.getLabel(), null);
300 		} else {
301 			writer.write(option.getLabel());
302 		}
303 
304 		writer.endElement("label");
305 	}

312 	/*
313 	 * return the value for the triState based on the value of the
314 	 * selectItems on the iteration
315 	 */
316 	protected int getValueForInput(FacesContext context, UIInput component, String itemValue, Map<String, Object> valueArray,
317 	                               Converter converter) {
318 		try {
319 			int retInt = 0;
320 			if (itemValue == null || valueArray == null) {
321 				return retInt;
322 			}
323 
324 			if (valueArray.containsKey(itemValue)) {
325 				retInt = Integer.parseInt(((String) valueArray.get(itemValue)));
326 
327 				return retInt % 3;
328 			} else {
329 				return retInt;
330 			}
331 		} catch (NumberFormatException ex) {
332 			throw new FacesException("State of '" + component.getClientId() + "' must be an integer representation");
333 		}
334 	}

336 	@Override
337     @SuppressWarnings("unchecked")
338 	protected Map<String, Object> getValues(UIComponent component) {
339 		UISelectMany selectMany = (UISelectMany) component;
340 		Object value = selectMany.getValue();
341 
342 		if (value == null) {
343 			return null;
344 		} else if (value instanceof Map) {
345 			//it should be a Map instance for <ItemStringValue,Value>
346 			return ((Map) value);
347 		} else {
348 			throw new FacesException("Value of '" + component.getClientId() + "'must be a Map instance");
349 		}
350 	}

352 	protected Map<String, Object> getSubmitedMap(FacesContext context, TriStateManyCheckbox checkbox, String[] valuesArray) {
353 		List<SelectItem> selectItems = getSelectItems(context, checkbox);
354 		Map<String, Object> subValues = new LinkedHashMap<String, Object>();
355 		if (valuesArray != null && valuesArray.length == selectItems.size()) {
356 			int idx = -1;
357 			for (SelectItem item : selectItems) {
358 				idx++;
359 
360 				String keyMap = (String) item.getValue();
361 				Object valueMap = valuesArray[idx];
362 				subValues.put(keyMap, valueMap);
363 			}
364 
365 			return subValues;
366 		} else {
367 			return null;
368 		}
369 	}

371 	protected Map<String, Object> getSubmittedFromComp(UIComponent component) {
372 		TriStateManyCheckbox checkbox = (TriStateManyCheckbox) component;
373 		@SuppressWarnings("unchecked")
374 		Map<String, Object> ret = (Map<String, Object>) checkbox.getSubmittedValue();
375 		if (ret != null) {
376 			Map<String, Object> subValues = new LinkedHashMap<String, Object>();
377 
378 			//need to reverse the order of element on the map to take the value as on decode.
379 			Set<String> keys = ret.keySet();
380 			String[] tempArray = (String[]) keys.toArray(new String[keys.size()]);
381 
382 			int length = tempArray.length;
383 			for (int i = length - 1; i >= 0; i--) {
384 				String key = tempArray[i];
385 				Object val = ret.get(key);
386 				subValues.put(key, val);
387 			}
388 
389 			return subValues;
390 		} else {
391 			return null;
392 		}
