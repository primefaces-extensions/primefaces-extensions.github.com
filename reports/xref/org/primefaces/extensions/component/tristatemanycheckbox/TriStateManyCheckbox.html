166     @Override
167     protected void validateValue(final FacesContext context, final Object value) {
168         final Map<Object, Object> mapValues = (Map) value;
169 
170         // call all validators
171         final Validator[] validators = getValidators();
172         if (getValidators() != null) {
173             for (final Validator validator : validators) {
174                 final Iterator<Object> it = mapValues.values().iterator();
175                 while (it.hasNext()) {
176                     final Object newValue = it.next();
177                     try {
178                         validator.validate(context, this, newValue);
179                     }
180                     catch (final ValidatorException ve) {
181                         // If the validator throws an exception, we're
182                         // invalid, and we need to add a message
183                         setValid(false);
184 
185                         FacesMessage message;
186                         final String validatorMessageString = getValidatorMessage();
187 
188                         if (null != validatorMessageString) {
189                             message = new FacesMessage(FacesMessage.SEVERITY_ERROR, validatorMessageString, validatorMessageString);
190                             message.setSeverity(FacesMessage.SEVERITY_ERROR);
191                         }
192                         else {
193                             final Collection<FacesMessage> messages = ve.getFacesMessages();
194                             if (null != messages) {
195                                 message = null;
196 
197                                 final String cid = getClientId(context);
198                                 for (final FacesMessage m : messages) {
199                                     context.addMessage(cid, m);
200                                 }
201                             }
202                             else {
203                                 message = ve.getFacesMessage();
204                             }
205                         }
206 
207                         if (message != null) {
208                             context.addMessage(getClientId(context), message);
209                         }
210                     }
211                 }
212             }
213         }
214 
215         boolean doAddMessage = false;
216 
217         // Ensure that if the state are all 0 and a
218         // value is required, a message is queued
219         if (isRequired()
220                     && isValid()) {
221             final Iterator<Object> it = mapValues.values().iterator();
222             boolean cCheck = true;
223             while (it.hasNext() && cCheck) {
224                 final Object val = it.next();
225                 if (!"0".equals(getConverter().getAsString(context, this, val))) {
226                     cCheck = false;
227                 }
228             }
229 
230             if (cCheck) {
231                 doAddMessage = true;
232             }
233         }
234 
235         if (doAddMessage) {
236             final Object[] params = new Object[2];
237             params[0] = MessageFactory.getLabel(context, this);
238 
239             // Enqueue an error message if an invalid value was specified
240             final FacesMessage message = MessageFactory.getMessage(UISelectMany.INVALID_MESSAGE_ID, FacesMessage.SEVERITY_ERROR, params);
241             context.addMessage(getClientId(context), message);
242             setValid(false);
243         }
244     }
