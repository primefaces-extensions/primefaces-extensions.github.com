221     @Test
222     void complexMap() {
223         jsonConverter.setType(
224                     "java.util.Map<java.lang.String, org.apache.commons.lang3.tuple.ImmutablePair<java.lang.Integer, java.util.Date>>");
225 
226         final Map<String, ImmutablePair<Integer, Date>> map = new HashMap<>();
227         GregorianCalendar calendar = new GregorianCalendar(2012, Calendar.FEBRUARY, 20);
228         map.put("cat", new ImmutablePair<>(1, calendar.getTime()));
229         calendar = new GregorianCalendar(2011, Calendar.JULY, 1);
230         map.put("dog", new ImmutablePair<>(2, calendar.getTime()));
231         calendar = new GregorianCalendar(1999, Calendar.NOVEMBER, 15);
232         map.put("unknow", new ImmutablePair<>(3, calendar.getTime()));
233 
234         final String json = jsonConverter.getAsString(null, null, map);
235 
236         final Object obj = jsonConverter.getAsObject(null, null, json);
237         assertTrue(CollectionUtils.isEqualCollection(map.entrySet(),
238                     ((Map<String, ImmutablePair<Integer, Date>>) obj).entrySet()));
239     }

266     @Test
267     void pojoGenericComplex() {
268         jsonConverter.setType(
269                     "org.primefaces.extensions.converter.FooGeneric<int[], "
270                                 + "org.primefaces.extensions.converter.FooGeneric<org.primefaces.extensions.converter.FooNonGeneric, java.lang.Boolean>>");
271 
272         final FooNonGeneric fooNonGeneric = new FooNonGeneric();
273         final FooGeneric<FooNonGeneric, Boolean> fooGenericInnner = new FooGeneric<>();
274         fooGenericInnner.setA(fooNonGeneric);
275         fooGenericInnner.setB(false);
276 
277         final FooGeneric<int[], FooGeneric<FooNonGeneric, Boolean>> fooGenericOuter = new FooGeneric<>();
278         final int[] ints = {1, 2, 3, 4, 5};
279         fooGenericOuter.setA(ints);
280         fooGenericOuter.setB(fooGenericInnner);
281 
282         final String json = jsonConverter.getAsString(null, null, fooGenericOuter);
283 
284         final Object obj = jsonConverter.getAsObject(null, null, json);
285         assertEquals(fooGenericOuter, obj);
